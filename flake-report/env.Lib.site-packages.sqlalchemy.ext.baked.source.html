133 def spoil(self, full=False):
134 """Cancel any query caching that will occur on this BakedQuery object.
135  
136 The BakedQuery can continue to be used normally, however additional
137 creational functions will not be cached; they will be called
138 on every invocation.
139  
140 This is to support the case where a particular step in constructing
141 a baked query disqualifies the query from being cacheable, such
142 as a variant that relies upon some uncacheable value.
143  
144 :param full: if False, only functions added to this
145 :class:`.BakedQuery` object subsequent to the spoil step will be
146 non-cached; the state of the :class:`.BakedQuery` up until
147 this point will be pulled from the cache. If True, then the
148 entire :class:`_query.Query` object is built from scratch each
149 time, with all creational functions being called on each
150 invocation.
151  
152 """
153 if not full and not self._spoiled:
154 _spoil_point = self._clone()
155 _spoil_point._cache_key += ("_query_only",)
156 self.steps = [_spoil_point._retrieve_baked_query]
157 self._spoiled = True
158 return self

182

185 "spoiled" so that it won't use caching.

189 key = ()

214 def _retrieve_baked_query(self, session):
215 query = self._bakery.get(self._effective_key(session), None)
216 if query is None:
217 query = self._as_query(session)
218 self._bakery[self._effective_key(session)] = query.with_session(
219 None

221 return query.with_session(session)

223 def _bake(self, session):
224 query = self._as_query(session)
225  
226 context = query._compile_context()
227  
228 self._bake_subquery_loaders(session, context)
229 context.session = None
230 context.query = query = context.query.with_session(None)
231 query._execution_options = query._execution_options.union(
232 {"compiled_cache": self._bakery}
233 )
234 # we'll be holding onto the query for some of its state,
235 # so delete some compilation-use-only attributes that can take up
236 # space
237 for attr in (
238 "_correlate",
239 "_from_obj",
240 "_mapper_adapter_map",
241 "_joinpath",
242 "_joinpoint",
243 ):
244 query.__dict__.pop(attr, None)
245  
246 # if the query is not safe to cache, we still do everything as though
247 # we did cache it, since the receiver of _bake() assumes subqueryload
248 # context was set up, etc.
249 if context.query._bake_ok:
250 self._bakery[self._effective_key(session)] = context
251  
252 return context

254 def to_query(self, query_or_session):
255 """Return the :class:`_query.Query` object for use as a subquery.
256  
257 This method should be used within the lambda callable being used
258 to generate a step of an enclosing :class:`.BakedQuery`. The
259 parameter should normally be the :class:`_query.Query` object that
260 is passed to the lambda::
261  
262 sub_bq = self.bakery(lambda s: s.query(User.name))
263 sub_bq += lambda q: q.filter(
264 User.id == Address.user_id).correlate(Address)
265  
266 main_bq = self.bakery(lambda s: s.query(Address))
267 main_bq += lambda q: q.filter(
268 sub_bq.to_query(q).exists())
269  
270 In the case where the subquery is used in the first callable against
271 a :class:`.Session`, the :class:`.Session` is also accepted::
272  
273 sub_bq = self.bakery(lambda s: s.query(User.name))
274 sub_bq += lambda q: q.filter(
275 User.id == Address.user_id).correlate(Address)
276  
277 main_bq = self.bakery(
278 lambda s: s.query(Address.id, sub_bq.to_query(q).as_scalar())
279 )
280  
281 :param query_or_session: a :class:`_query.Query` object or a class
282 :class:`.Session` object, that is assumed to be within the context
283 of an enclosing :class:`.BakedQuery` callable.
284  
285  
286 .. versionadded:: 1.3
287  
288  
289 """
290  
291 if isinstance(query_or_session, Session):
292 session = query_or_session
293 elif isinstance(query_or_session, Query):
294 session = query_or_session.session
295 if session is None:
296 raise sa_exc.ArgumentError(
297 "Given Query needs to be associated with a Session"
298 )
299 else:
300 raise TypeError(
301 "Query or Session object expected, got %r."
302 % type(query_or_session)
303 )
304 return self._as_query(session)

313 def _bake_subquery_loaders(self, session, context):
314 """convert subquery eager loaders in the cache into baked queries.
315  
316 For subquery eager loading to work, all we need here is that the
317 Query point to the correct session when it is run. However, since
318 we are "baking" anyway, we may as well also turn the query into
319 a "baked" query so that we save on performance too.
320  
321 """
322 context.attributes["baked_queries"] = baked_queries = []
323 for k, v in list(context.attributes.items()):
324 if isinstance(v, Query):
325 if "subquery" in k:
326 bk = BakedQuery(self._bakery, lambda *args: v)
327 bk._cache_key = self._cache_key + k
328 bk._bake(session)
329 baked_queries.append((k, bk._cache_key, v))
330 del context.attributes[k]
331  
332 def _unbake_subquery_loaders(
333 self, session, context, params, post_criteria
334 ):
335 """Retrieve subquery eager loaders stored by _bake_subquery_loaders
336 and turn them back into Result objects that will iterate just
337 like a Query object.
338  
339 """
340 if "baked_queries" not in context.attributes:
341 return
342  
343 for k, cache_key, query in context.attributes["baked_queries"]:
344 bk = BakedQuery(
345 self._bakery, lambda sess, q=query: q.with_session(sess)
346 )
347 bk._cache_key = cache_key
348 q = bk.for_session(session)
349 for fn in post_criteria:
350 q = q.with_post_criteria(fn)
351 context.attributes[k] = q.params(**params)

389 def with_post_criteria(self, fn):
390 """Add a criteria function that will be applied post-cache.
391  
392 This adds a function that will be run against the
393 :class:`_query.Query` object after it is retrieved from the
394 cache. Functions here can be used to alter the query in ways
395 that **do not affect the SQL output**, such as execution options
396 and shard identifiers (when using a shard-enabled query object)
397  
398 .. warning:: :meth:`.Result.with_post_criteria` functions are applied
399 to the :class:`_query.Query`
400 object **after** the query's SQL statement
401 object has been retrieved from the cache. Any operations here
402 which intend to modify the SQL should ensure that
403 :meth:`.BakedQuery.spoil` was called first.
404  
405 .. versionadded:: 1.2
406  
407  
408 """
409 return self._using_post_criteria([fn])

411 def _as_query(self):
412 q = self.bq._as_query(self.session).params(self._params)
413 for fn in self._post_criteria:
414 q = fn(q)
415 return q
