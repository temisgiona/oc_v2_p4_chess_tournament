75 def merge(self, tokens):
76 """Merge subsequent tokens of same token-type.
77  
78 Also strip the final newline (added by pygments).
79 """
80 tokens = iter(tokens)
81 (lasttype, lastval) = next(tokens)
82 for ttype, value in tokens:
83 if ttype is lasttype:
84 lastval += value
85 else:
86 yield(lasttype, lastval)
87 (lasttype, lastval) = (ttype, value)
88 if lastval.endswith('\n'):
89 lastval = lastval[:-1]
90 if lastval:
91 yield(lasttype, lastval)

111
