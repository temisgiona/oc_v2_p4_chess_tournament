101 def test_pyarrow_array_to_numpy_and_mask(np_dtype_to_arrays):
102 """
103 Test conversion from pyarrow array to numpy array.
104  
105 Modifies the pyarrow buffer to contain padding and offset, which are
106 considered valid buffers by pyarrow.
107  
108 Also tests empty pyarrow arrays with non empty buffers.
109 See https://github.com/pandas-dev/pandas/issues/40896
110 """
111 np_dtype, pa_array, np_expected, mask_expected = np_dtype_to_arrays
112 data, mask = pyarrow_array_to_numpy_and_mask(pa_array, np_dtype)
113 tm.assert_numpy_array_equal(data[:3], np_expected)
114 tm.assert_numpy_array_equal(mask, mask_expected)
115  
116 mask_buffer = pa_array.buffers()[0]
117 data_buffer = pa_array.buffers()[1]
118 data_buffer_bytes = pa_array.buffers()[1].to_pybytes()
119  
120 # Add trailing padding to the buffer.
121 data_buffer_trail = pa.py_buffer(data_buffer_bytes + b"\x00")
122 pa_array_trail = pa.Array.from_buffers(
123 type=pa_array.type,
124 length=len(pa_array),
125 buffers=[mask_buffer, data_buffer_trail],
126 offset=pa_array.offset,
127 )
128 pa_array_trail.validate()
129 data, mask = pyarrow_array_to_numpy_and_mask(pa_array_trail, np_dtype)
130 tm.assert_numpy_array_equal(data[:3], np_expected)
131 tm.assert_numpy_array_equal(mask, mask_expected)
132  
133 # Add offset to the buffer.
134 offset = b"\x00" * (pa_array.type.bit_width // 8)
135 data_buffer_offset = pa.py_buffer(offset + data_buffer_bytes)
136 mask_buffer_offset = pa.py_buffer(b"\x0E")
137 pa_array_offset = pa.Array.from_buffers(
138 type=pa_array.type,
139 length=len(pa_array),
140 buffers=[mask_buffer_offset, data_buffer_offset],
141 offset=pa_array.offset + 1,
142 )
143 pa_array_offset.validate()
144 data, mask = pyarrow_array_to_numpy_and_mask(pa_array_offset, np_dtype)
145 tm.assert_numpy_array_equal(data[:3], np_expected)
146 tm.assert_numpy_array_equal(mask, mask_expected)
