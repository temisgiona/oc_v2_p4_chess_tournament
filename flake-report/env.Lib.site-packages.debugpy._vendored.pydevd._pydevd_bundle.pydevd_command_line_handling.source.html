122 """ parses the arguments.
123 removes our arguments from the command line """
124 setup = {}
125 for handler in ACCEPTED_ARG_HANDLERS:

133 handler = ARGV_REP_TO_HANDLER.get(argv[i])
134 if handler is not None:
135 handler.handle_argv(argv, i, setup)
136  
137 elif argv[i].startswith('--qt-support'):
138 # The --qt-support is special because we want to keep backward compatibility:
139 # Previously, just passing '--qt-support' meant that we should use the auto-discovery mode
140 # whereas now, if --qt-support is passed, it should be passed as --qt-support=<mode>, where
141 # mode can be one of 'auto', 'none', 'pyqt5', 'pyqt4', 'pyside', 'pyside2'.
142 if argv[i] == '--qt-support':
143 setup['qt-support'] = 'auto'
144  
145 elif argv[i].startswith('--qt-support='):
146 qt_support = argv[i][len('--qt-support='):]
147 valid_modes = ('none', 'auto', 'pyqt5', 'pyqt4', 'pyside', 'pyside2')
148 if qt_support not in valid_modes:
149 raise ValueError("qt-support mode invalid: " + qt_support)
150 if qt_support == 'none':
151 # On none, actually set an empty string to evaluate to False.
152 setup['qt-support'] = ''
153 else:
154 setup['qt-support'] = qt_support
155 else:
156 raise ValueError("Unexpected definition for qt-support flag: " + argv[i])
157  
158 del argv[i]
159  
160 elif argv[i] == '--file':
161 # --file is special because it's the last one (so, no handler for it).
162 del argv[i]

164 i = len(argv) # pop out, file is our last argument

166 elif argv[i] == '--DEBUG':

173 return setup
