8 """Support for collections of mapped entities.
9  
10 The collections package supplies the machinery used to inform the ORM of
11 collection membership changes. An instrumentation via decoration approach is
12 used, allowing arbitrary types (including built-ins) to be used as entity
13 collections without requiring inheritance from a base class.
14  
15 Instrumentation decoration relays membership change events to the
16 :class:`.CollectionAttributeImpl` that is currently managing the collection.
17 The decorators observe function call arguments and return values, tracking
18 entities entering or leaving the collection. Two decorator approaches are
19 provided. One is a bundle of generic decorators that map function arguments
20 and return values to events::
21  
22 from sqlalchemy.orm.collections import collection
23 class MyClass(object):
24 # ...
25  
26 @collection.adds(1)
27 def store(self, item):
28 self.data.append(item)
29  
30 @collection.removes_return()
31 def pop(self):
32 return self.data.pop()
33  
34  
35 The second approach is a bundle of targeted decorators that wrap appropriate
36 append and remove notifiers around the mutation methods present in the
37 standard Python ``list``, ``set`` and ``dict`` interfaces. These could be
38 specified in terms of generic decorator recipes, but are instead hand-tooled
39 for increased efficiency. The targeted decorators occasionally implement
40 adapter-like behavior, such as mapping bulk-set methods (``extend``,
41 ``update``, ``__setslice__``, etc.) into the series of atomic mutation events
42 that the ORM requires.
43  
44 The targeted decorators are used internally for automatic instrumentation of
45 entity collection classes. Every collection class goes through a
46 transformation process roughly like so:
47  
48 1. If the class is a built-in, substitute a trivial sub-class
49 2. Is this class already instrumented?
50 3. Add in generic decorators
51 4. Sniff out the collection interface through duck-typing
52 5. Add targeted decoration to any undecorated interface method
53  
54 This process modifies the class at runtime, decorating methods and adding some
55 bookkeeping properties. This isn't possible (or desirable) for built-in
56 classes like ``list``, so trivial sub-classes are substituted to hold
57 decoration::
58  
59 class InstrumentedList(list):
60 pass
61  
62 Collection classes can be specified in ``relationship(collection_class=)`` as
63 types or a function that returns an instance. Collection classes are
64 inspected and instrumented during the mapper compilation phase. The
65 collection_class callable will be executed once to produce a specimen
66 instance, and the type of that specimen will be instrumented. Functions that
67 return built-in types like ``lists`` will be adapted to produce instrumented
68 instances.
69  
70 When extending a known type like ``list``, additional decorations are not
71 generally not needed. Odds are, the extension method will delegate to a
72 method that's already instrumented. For example::
73  
74 class QueueIsh(list):
75 def push(self, item):
76 self.append(item)
77 def shift(self):
78 return self.pop(0)
79  
80 There's no need to decorate these methods. ``append`` and ``pop`` are already
81 instrumented as part of the ``list`` interface. Decorating them would fire
82 duplicate events, which should be avoided.
83  
84 The targeted decoration tries not to rely on other methods in the underlying
85 collection class, but some are unavoidable. Many depend on 'read' methods
86 being present to properly instrument a 'write', for example, ``__setitem__``
87 needs ``__getitem__``. "Bulk" methods like ``update`` and ``extend`` may also
88 reimplemented in terms of atomic appends and removes, so the ``extend``
89 decoration will actually perform many ``append`` operations and not call the
90 underlying method at all.
91  
92 Tight control over bulk operation and the firing of events is also possible by
93 implementing the instrumentation internally in your methods. The basic
94 instrumentation package works under the general assumption that collection
95 mutation will not raise unusual exceptions. If you want to closely
96 orchestrate append and remove events with exception management, internal
97 instrumentation may be the answer. Within your method,
98 ``collection_adapter(self)`` will retrieve an object that you can use for
99 explicit control over triggering append and remove events.
100  
101 The owning object and :class:`.CollectionAttributeImpl` are also reachable
102 through the adapter, allowing for some very sophisticated behavior.
103  
104 """

191 class _SerializableColumnGetterV2(_PlainColumnGetter):
192 """Updated serializable getter which deals with
193 multi-table mapped classes.
194  
195 Two extremely unusual cases are not supported.
196 Mappings which have tables across multiple metadata
197 objects, or which are mapped to non-Table selectables
198 linked across inheriting mappers may fail to function
199 here.
200  
201 """
202  
203 def __init__(self, colkeys):
204 self.colkeys = colkeys
205 self.composite = len(colkeys) > 1
206  
207 def __reduce__(self):
208 return self.__class__, (self.colkeys,)
209  
210 @classmethod
211 def _reduce_from_cols(cls, cols):
212 def _table_key(c):
213 if not isinstance(c.table, expression.TableClause):
214 return None
215 else:
216 return c.table.key
217  
218 colkeys = [(c.key, _table_key(c)) for c in cols]
219 return _SerializableColumnGetterV2, (colkeys,)
220  
221 def _cols(self, mapper):
222 cols = []
223 metadata = getattr(mapper.local_table, "metadata", None)
224 for (ckey, tkey) in self.colkeys:
225 if tkey is None or metadata is None or tkey not in metadata:
226 cols.append(mapper.local_table.c[ckey])
227 else:
228 cols.append(metadata.tables[tkey].c[ckey])
229 return cols

301 class collection(object):
302 """Decorators for entity collection classes.
303  
304 The decorators fall into two groups: annotations and interception recipes.
305  
306 The annotating decorators (appender, remover, iterator, linker, converter,
307 internally_instrumented) indicate the method's purpose and take no
308 arguments. They are not written with parens::
309  
310 @collection.appender
311 def append(self, append): ...
312  
313 The recipe decorators all require parens, even those that take no
314 arguments::
315  
316 @collection.adds('entity')
317 def insert(self, position, entity): ...
318  
319 @collection.removes_return()
320 def popitem(self): ...
321  
322 """
323  
324 # Bundled as a class solely for ease of use: packaging, doc strings,
325 # importability.
326  
327 @staticmethod
328 def appender(fn):
329 """Tag the method as the collection appender.
330  
331 The appender method is called with one positional argument: the value
332 to append. The method will be automatically decorated with 'adds(1)'
333 if not already decorated::
334  
335 @collection.appender
336 def add(self, append): ...
337  
338 # or, equivalently
339 @collection.appender
340 @collection.adds(1)
341 def add(self, append): ...
342  
343 # for mapping type, an 'append' may kick out a previous value
344 # that occupies that slot. consider d['a'] = 'foo'- any previous
345 # value in d['a'] is discarded.
346 @collection.appender
347 @collection.replaces(1)
348 def add(self, entity):
349 key = some_key_func(entity)
350 previous = None
351 if key in self:
352 previous = self[key]
353 self[key] = entity
354 return previous
355  
356 If the value to append is not allowed in the collection, you may
357 raise an exception. Something to remember is that the appender
358 will be called for each object mapped by a database query. If the
359 database contains rows that violate your collection semantics, you
360 will need to get creative to fix the problem, as access via the
361 collection will not work.
362  
363 If the appender method is internally instrumented, you must also
364 receive the keyword argument '_sa_initiator' and ensure its
365 promulgation to collection events.
366  
367 """
368 fn._sa_instrument_role = "appender"
369 return fn
370  
371 @staticmethod
372 def remover(fn):
373 """Tag the method as the collection remover.
374  
375 The remover method is called with one positional argument: the value
376 to remove. The method will be automatically decorated with
377 :meth:`removes_return` if not already decorated::
378  
379 @collection.remover
380 def zap(self, entity): ...
381  
382 # or, equivalently
383 @collection.remover
384 @collection.removes_return()
385 def zap(self, ): ...
386  
387 If the value to remove is not present in the collection, you may
388 raise an exception or return None to ignore the error.
389  
390 If the remove method is internally instrumented, you must also
391 receive the keyword argument '_sa_initiator' and ensure its
392 promulgation to collection events.
393  
394 """
395 fn._sa_instrument_role = "remover"
396 return fn

401  
402 The iterator method is called with no arguments. It is expected to
403 return an iterator over all collection members::
404  
405 @collection.iterator
406 def __iter__(self): ...

409 fn._sa_instrument_role = "iterator"

471 def converter(fn):
472 """Tag the method as the collection converter.
473  
474 This optional method will be called when a collection is being
475 replaced entirely, as in::
476  
477 myobj.acollection = [newvalue1, newvalue2]
478  
479 The converter method will receive the object being assigned and should
480 return an iterable of values suitable for use by the ``appender``
481 method. A converter must not assign values or mutate the collection,
482 its sole job is to adapt the value the user provides into an iterable
483 of values for the ORM's use.
484  
485 The default converter implementation will use duck-typing to do the
486 conversion. A dict-like collection will be convert into an iterable
487 of dictionary values, and other types will simply be iterated::
488  
489 @collection.converter
490 def convert(self, other): ...
491  
492 If the duck-typing of the object does not match the type of this
493 collection, a TypeError is raised.
494

496 range of possible types that can be assigned in bulk or perform
497 validation on the values about to be assigned.
498  
499 """

607 entity collections.
608

1582 class MappedCollection(dict):
1583 """A basic dictionary-based collection class.
1584  
1585 Extends dict with the minimal bag semantics that collection
1586 classes require. ``set`` and ``remove`` are implemented in terms
1587 of a keying function: any callable that takes an object and
1588 returns an object for use as a dictionary key.
1589  
1590 """
1591  
1592 def __init__(self, keyfunc):
1593 """Create a new collection with keying provided by keyfunc.
1594  
1595 keyfunc may be any callable that takes an object and returns an object
1596 for use as a dictionary key.
1597  
1598 The keyfunc will be called every time the ORM needs to add a member by
1599 value-only (such as when loading instances from the database) or
1600 remove a member. The usual cautions about dictionary keying apply-
1601 ``keyfunc(object)`` should return the same output for the life of the
1602 collection. Keying based on mutable properties can result in
1603 unreachable instances "lost" in the collection.
1604  
1605 """
1606 self.keyfunc = keyfunc
1607  
1608 @collection.appender
1609 @collection.internally_instrumented
1610 def set(self, value, _sa_initiator=None):
1611 """Add an item by value, consulting the keyfunc for the key."""
1612  
1613 key = self.keyfunc(value)
1614 self.__setitem__(key, value, _sa_initiator)
1615  
1616 @collection.remover
1617 @collection.internally_instrumented
1618 def remove(self, value, _sa_initiator=None):
1619 """Remove an item by value, consulting the keyfunc for the key."""
1620  
1621 key = self.keyfunc(value)
1622 # Let self[key] raise if key is not in this collection
1623 # testlib.pragma exempt:__ne__
1624 if self[key] != value:
1625 raise sa_exc.InvalidRequestError(
1626 "Can not remove '%s': collection holds '%s' for key '%s'. "
1627 "Possible cause: is the MappedCollection key function "
1628 "based on mutable properties or properties that only obtain "
1629 "values after flush?" % (value, self[key], key)
1630 )
1631 self.__delitem__(key, _sa_initiator)
