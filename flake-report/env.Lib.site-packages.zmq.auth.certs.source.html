85 def load_certificate(filename):
86 """Load public and secret key from a zmq certificate.
87  
88 Returns (public_key, secret_key)
89  
90 If the certificate file only contains the public key,
91 secret_key will be None.
92  
93 If there is no public key found in the file, ValueError will be raised.
94 """
95 public_key = None
96 secret_key = None
97 if not os.path.exists(filename):
98 raise IOError("Invalid certificate file: {0}".format(filename))
99  
100 with open(filename, 'rb') as f:
101 for line in f:
102 line = line.strip()
103 if line.startswith(b'#'):
104 continue
105 if line.startswith(b'public-key'):
106 public_key = line.split(b"=", 1)[1].strip(b' \t\'"')
107 if line.startswith(b'secret-key'):
108 secret_key = line.split(b"=", 1)[1].strip(b' \t\'"')
109 if public_key and secret_key:
110 break
111  
112 if public_key is None:
113 raise ValueError("No public key found in %s" % filename)
114  
115 return public_key, secret_key

118 def load_certificates(directory='.'):
119 """Load public keys from all certificates in a directory"""
120 certs = {}
121 if not os.path.isdir(directory):
122 raise IOError("Invalid certificate directory: {0}".format(directory))
123 # Follow czmq pattern of public keys stored in *.key files.
124 glob_string = os.path.join(directory, "*.key")
125  
126 cert_files = glob.glob(glob_string)
127 for cert_file in cert_files:
128 public_key, _ = load_certificate(cert_file)
129 if public_key:
130 certs[public_key] = True
131 return certs
