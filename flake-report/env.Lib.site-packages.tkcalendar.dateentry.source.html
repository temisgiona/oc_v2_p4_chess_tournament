84 def __init__(self, master=None, **kw):
85 """
86 Create an entry with a drop-down calendar to select a date.
87  
88 When the entry looses focus, if the user input is not a valid date,
89 the entry content is reset to the last valid date.
90  
91 Keyword Options
92 ---------------
93  
94 usual ttk.Entry options and Calendar options.
95 The Calendar option 'cursor' has been renamed
96 'calendar_cursor' to avoid name clashes with the
97 corresponding ttk.Entry option.
98  
99 Virtual event
100 -------------
101  
102 A ``<<DateEntrySelected>>`` event is generated each time
103 the user selects a date.
104  
105 """
106 # sort keywords between entry options and calendar options
107 kw['selectmode'] = 'day'
108 entry_kw = {}
109  
110 style = kw.pop('style', 'DateEntry')
111  
112 for key in self.entry_kw:
113 entry_kw[key] = kw.pop(key, self.entry_kw[key])
114 entry_kw['font'] = kw.get('font', None)
115 self._cursor = entry_kw['cursor'] # entry cursor
116 kw['cursor'] = kw.pop('calendar_cursor', None)
117  
118 ttk.Entry.__init__(self, master, **entry_kw)
119  
120 self._determine_downarrow_name_after_id = ''
121  
122 # drop-down calendar
123 self._top_cal = tk.Toplevel(self)
124 self._top_cal.withdraw()
125 if platform == "linux":
126 self._top_cal.attributes('-type', 'DROPDOWN_MENU')
127 self._top_cal.overrideredirect(True)
128 self._calendar = Calendar(self._top_cal, **kw)
129 self._calendar.pack()
130  
131 # locale date parsing / formatting
132 self.format_date = self._calendar.format_date
133 self.parse_date = self._calendar.parse_date
134  
135 # style
136 self._theme_name = '' # to detect theme changes
137 self.style = ttk.Style(self)
138 self._setup_style()
139 self.configure(style=style)
140  
141 # add validation to Entry so that only dates in the locale's format
142 # are accepted
143 validatecmd = self.register(self._validate_date)
144 self.configure(validate='focusout',
145 validatecommand=validatecmd)
146  
147 # initially selected date
148 self._date = self._calendar.selection_get()
149 if self._date is None:
150 today = self._calendar.date.today()
151 year = kw.get('year', today.year)
152 month = kw.get('month', today.month)
153 day = kw.get('day', today.day)
154 try:
155 self._date = self._calendar.date(year, month, day)
156 except ValueError:
157 self._date = today
158 self._set_text(self.format_date(self._date))
159  
160 # --- bindings
161 # reconfigure style if theme changed
162 self.bind('<<ThemeChanged>>',
163 lambda e: self.after(10, self._on_theme_change))
164 # determine new downarrow button bbox
165 self.bind('<Configure>', self._determine_downarrow_name)
166 self.bind('<Map>', self._determine_downarrow_name)
167 # handle appearence to make the entry behave like a Combobox but with
168 # a drop-down calendar instead of a drop-down list
169 self.bind('<Leave>', lambda e: self.state(['!active']))
170 self.bind('<Motion>', self._on_motion)
171 self.bind('<ButtonPress-1>', self._on_b1_press)
172 # update entry content when date is selected in the Calendar
173 self._calendar.bind('<<CalendarSelected>>', self._select)
174 # hide calendar if it looses focus
175 self._calendar.bind('<FocusOut>', self._on_focus_out_cal)

206 def _determine_downarrow_name(self, event=None):
207 """Determine downarrow button name."""
208 try:
209 self.after_cancel(self._determine_downarrow_name_after_id)
210 except ValueError:
211 # nothing to cancel
212 pass
213 if self.winfo_ismapped():
214 self.update_idletasks()
215 y = self.winfo_height() // 2
216 x = self.winfo_width() - 10
217 name = self.identify(x, y)
218 if name:
219 self._downarrow_name = name
220 else:
221 self._determine_downarrow_name_after_id = self.after(10, self._determine_downarrow_name)

223 def _on_motion(self, event):
224 """Set widget state depending on mouse position to mimic Combobox behavior."""
225 x, y = event.x, event.y
226 if 'disabled' not in self.state():
227 if self.identify(x, y) == self._downarrow_name:
228 self.state(['active'])
229 ttk.Entry.configure(self, cursor='arrow')
230 else:
231 self.state(['!active'])
232 ttk.Entry.configure(self, cursor=self._cursor)

241 def _on_b1_press(self, event):
242 """Trigger self.drop_down on downarrow button press and set widget state to ['pressed', 'active']."""
243 x, y = event.x, event.y
244 if (('disabled' not in self.state()) and self.identify(x, y) == self._downarrow_name):
245 self.state(['pressed'])

248 def _on_focus_out_cal(self, event):
249 """Withdraw drop-down calendar when it looses focus."""
250 if self.focus_get() is not None:
251 if self.focus_get() == self:
252 x, y = event.x, event.y
253 if (type(x) != int or type(y) != int or self.identify(x, y) != self._downarrow_name):
254 self._top_cal.withdraw()
255 self.state(['!pressed'])
256 else:
257 self._top_cal.withdraw()
258 self.state(['!pressed'])
259 elif self.grab_current():
260 # 'active' won't be in state because of the grab
261 x, y = self._top_cal.winfo_pointerxy()
262 xc = self._top_cal.winfo_rootx()
263 yc = self._top_cal.winfo_rooty()
264 w = self._top_cal.winfo_width()
265 h = self._top_cal.winfo_height()
266 if xc <= x <= xc + w and yc <= y <= yc + h:
267 # re-focus calendar so that <FocusOut> will be triggered next time
268 self._calendar.focus_force()
269 else:
270 self._top_cal.withdraw()
271 self.state(['!pressed'])
272 else:
273 if 'active' in self.state():
274 # re-focus calendar so that <FocusOut> will be triggered next time
275 self._calendar.focus_force()
276 else:
277 self._top_cal.withdraw()
278 self.state(['!pressed'])

280 def _validate_date(self):
281 """Date entry validation: only dates in locale '%x' format are accepted."""
282 try:
283 date = self.parse_date(self.get())
284 self._date = self._calendar.check_date_range(date)
285 if self._date != date:
286 self._set_text(self.format_date(self._date))
287 return False
288 else:
289 return True
290 except (ValueError, IndexError):
291 self._set_text(self.format_date(self._date))
292 return False

294 def _select(self, event=None):
295 """Display the selected date in the entry and hide the calendar."""
296 date = self._calendar.selection_get()
297 if date is not None:
298 self._set_text(self.format_date(date))
299 self._date = date
300 self.event_generate('<<DateEntrySelected>>')
301 self._top_cal.withdraw()
302 if 'readonly' not in self.state():
303 self.focus_set()

305 def _set_text(self, txt):
306 """Insert text in the entry."""
307 if 'readonly' in self.state():
308 readonly = True
309 self.state(('!readonly',))
310 else:
311 readonly = False
312 self.delete(0, 'end')
313 self.insert(0, txt)
314 if readonly:
315 self.state(('readonly',))

325 def drop_down(self):
326 """Display or withdraw the drop-down calendar depending on its current state."""
327 if self._calendar.winfo_ismapped():
328 self._top_cal.withdraw()
329 else:
330 self._validate_date()
331 date = self.parse_date(self.get())
332 x = self.winfo_rootx()
333 y = self.winfo_rooty() + self.winfo_height()
334 if self.winfo_toplevel().attributes('-topmost'):
335 self._top_cal.attributes('-topmost', True)
336 else:
337 self._top_cal.attributes('-topmost', False)
338 self._top_cal.geometry('+%i+%i' % (x, y))
339 self._top_cal.deiconify()
340 self._calendar.focus_set()
341 self._calendar.selection_set(date)

356 self.configure(cursor='arrow')
357 elif '!disabled' in states or '!readonly' in states:

411 def set_date(self, date):
412 """
413 Set the value of the DateEntry to date.
414  
415 date can be a datetime.date, a datetime.datetime or a string
416 in locale '%x' format.
417 """
418 try:
419 txt = self.format_date(date)
420 except AssertionError:
421 txt = str(date)
422 try:
423 self.parse_date(txt)
424 except Exception:
425 raise ValueError("%r is not a valid date." % date)
426 self._set_text(txt)
427 self._validate_date()
