134 def parse_list(
135 parse_item: Callable[[str, int, str], Tuple[T, int]],
136 header: str,
137 pos: int,
138 header_name: str,
139 ) -> List[T]:
140 """
141 Parse a comma-separated list from ``header`` at the given position.
142  
143 This is appropriate for parsing values with the following grammar:
144  
145 1#item
146  
147 ``parse_item`` parses one item.
148  
149 ``header`` is assumed not to start or end with whitespace.
150  
151 (This function is designed for parsing an entire header value and
152 :func:`~websockets.http.read_headers` strips whitespace from values.)
153  
154 Return a list of items.
155  
156 :raises ~websockets.exceptions.InvalidHeaderFormat: on invalid inputs.
157  
158 """
159 # Per https://tools.ietf.org/html/rfc7230#section-7, "a recipient MUST
160 # parse and ignore a reasonable number of empty list elements"; hence
161 # while loops that remove extra delimiters.
162  
163 # Remove extra delimiters before the first item.
164 while peek_ahead(header, pos) == ",":
165 pos = parse_OWS(header, pos + 1)
166  
167 items = []
168 while True:
169 # Loop invariant: a item starts at pos in header.
170 item, pos = parse_item(header, pos, header_name)
171 items.append(item)
172 pos = parse_OWS(header, pos)
173  
174 # We may have reached the end of the header.
175 if pos == len(header):
176 break
177  
178 # There must be a delimiter after each element except the last one.
179 if peek_ahead(header, pos) == ",":
180 pos = parse_OWS(header, pos + 1)
181 else:
  • E501 Line too long (81 > 79 characters)
182 raise InvalidHeaderFormat(header_name, "expected comma", header, pos)
183  
184 # Remove extra delimiters before the next item.
185 while peek_ahead(header, pos) == ",":
186 pos = parse_OWS(header, pos + 1)
187  
188 # We may have reached the end of the header.
189 if pos == len(header):
190 break
191  
192 # Since we only advance in the header by one character with peek_ahead()
193 # or with the end position of a regex match, we can't overshoot the end.
194 assert pos == len(header)
195  
196 return items

263 header: str, pos: int, header_name: str

265 """
266 Parse a single extension parameter from ``header`` at the given position.
267

269  
270 :raises ~websockets.exceptions.InvalidHeaderFormat: on invalid inputs.
271

  • E501 Line too long (84 > 79 characters)

290 value, pos = parse_token(header, pos, header_name)
291 pos = parse_OWS(header, pos)

296 def parse_extension_item(
297 header: str, pos: int, header_name: str
298 ) -> Tuple[ExtensionHeader, int]:
299 """
300 Parse an extension definition from ``header`` at the given position.
301  
302 Return an ``(extension name, parameters)`` pair, where ``parameters`` is a
303 list of ``(name, value)`` pairs, and the new position.
304  
305 :raises ~websockets.exceptions.InvalidHeaderFormat: on invalid inputs.
306  
307 """
308 # Extract extension name.
309 name, pos = parse_token(header, pos, header_name)
310 pos = parse_OWS(header, pos)
311 # Extract all parameters.
312 parameters = []

315 parameter, pos = parse_extension_item_param(header, pos, header_name)
316 parameters.append(parameter)
317 return (cast(ExtensionName, name), parameters), pos

467 Parse an ``Authorization`` header for HTTP Basic Auth.
468  
469 Return a ``(username, password)`` tuple.

471 :param header: value of the ``Authorization`` header
472 :raises InvalidHeaderFormat: on invalid inputs
473 :raises InvalidHeaderValue: on unsupported inputs

477 # https://tools.ietf.org/html/rfc7617#section-2
