27 class RaiseExcInfoTest(unittest.TestCase):
28 def test_two_arg_exception(self):
29 # This test would fail on python 3 if raise_exc_info were simply
30 # a three-argument raise statement, because TwoArgException
31 # doesn't have a "copy constructor"
32 class TwoArgException(Exception):
33 def __init__(self, a, b):
34 super().__init__()
35 self.a, self.b = a, b
36  
37 try:
38 raise TwoArgException(1, 2)
39 except TwoArgException:
40 exc_info = sys.exc_info()
41 try:
42 raise_exc_info(exc_info)
43 self.fail("didn't get expected exception")
44 except TwoArgException as e:
45 self.assertIs(e, exc_info[1])
