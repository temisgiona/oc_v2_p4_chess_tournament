887 def _interp_limit(invalid: np.ndarray, fw_limit, bw_limit):
888 """
889 Get indexers of values that won't be filled
890 because they exceed the limits.
891  
892 Parameters
893 ----------
894 invalid : np.ndarray[bool]
895 fw_limit : int or None
896 forward limit to index
897 bw_limit : int or None
898 backward limit to index
899  
900 Returns
901 -------
902 set of indexers
903  
904 Notes
905 -----
906 This is equivalent to the more readable, but slower
907  
908 .. code-block:: python
909  
910 def _interp_limit(invalid, fw_limit, bw_limit):
911 for x in np.where(invalid)[0]:
912 if invalid[max(0, x - fw_limit):x + bw_limit + 1].all():
913 yield x
914 """
915 # handle forward first; the backward direction is the same except
916 # 1. operate on the reversed array
917 # 2. subtract the returned indices from N - 1
918 N = len(invalid)
919 f_idx = set()
920 b_idx = set()
921  
922 def inner(invalid, limit):
923 limit = min(limit, N)
924 windowed = _rolling_window(invalid, limit + 1).all(1)
925 idx = set(np.where(windowed)[0] + limit) | set(
926 np.where((~invalid[: limit + 1]).cumsum() == 0)[0]
927 )
928 return idx
929  
930 if fw_limit is not None:
931  
932 if fw_limit == 0:
933 f_idx = set(np.where(invalid)[0])
934 else:
935 f_idx = inner(invalid, fw_limit)
936  
937 if bw_limit is not None:
938  
939 if bw_limit == 0:
940 # then we don't even need to care about backwards
941 # just use forwards
942 return f_idx
943 else:
944 b_idx_inv = list(inner(invalid[::-1], bw_limit))
945 b_idx = set(N - 1 - np.asarray(b_idx_inv))
946 if fw_limit == 0:
947 return b_idx
948  
949 return f_idx & b_idx

952 def _rolling_window(a: np.ndarray, window: int):
953 """
954 [True, True, False, True, False], 2 ->
955  
956 [
957 [True, True],
958 [True, False],
959 [False, True],
960 [True, False],
961 ]
962 """
963 # https://stackoverflow.com/a/6811241
964 shape = a.shape[:-1] + (a.shape[-1] - window + 1, window)
965 strides = a.strides + (a.strides[-1],)
966 return np.lib.stride_tricks.as_strided(a, shape=shape, strides=strides)
