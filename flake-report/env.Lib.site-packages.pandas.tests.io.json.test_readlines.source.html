268 class MyReader:
269 def __init__(self, contents):
270 self.read_count = 0
271 self.stringio = StringIO(contents)
272  
273 def read(self, *args):
274 self.read_count += 1
275 return self.stringio.read(*args)
276  
277 def __iter__(self):
278 self.read_count += 1
279 return iter(self.stringio)
280
