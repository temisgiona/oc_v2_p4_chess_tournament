77 def _masked_arith_op(x: np.ndarray, y, op):
78 """
79 If the given arithmetic operation fails, attempt it again on
80 only the non-null elements of the input array(s).
81  
82 Parameters
83 ----------
84 x : np.ndarray
85 y : np.ndarray, Series, Index
86 op : binary operator
87 """
88 # For Series `x` is 1D so ravel() is a no-op; calling it anyway makes
89 # the logic valid for both Series and DataFrame ops.
90 xrav = x.ravel()
91 assert isinstance(x, np.ndarray), type(x)
92 if isinstance(y, np.ndarray):
93 dtype = find_common_type([x.dtype, y.dtype])
94 # error: Argument "dtype" to "empty" has incompatible type
95 # "Union[dtype, ExtensionDtype]"; expected "Union[dtype, None, type,
96 # _SupportsDtype, str, Tuple[Any, int], Tuple[Any, Union[int,
97 # Sequence[int]]], List[Any], _DtypeDict, Tuple[Any, Any]]"
98 result = np.empty(x.size, dtype=dtype) # type: ignore[arg-type]
99  
100 if len(x) != len(y):
101 raise ValueError(x.shape, y.shape)
102 else:
103 ymask = notna(y)
104  
105 # NB: ravel() is only safe since y is ndarray; for e.g. PeriodIndex
106 # we would get int64 dtype, see GH#19956
107 yrav = y.ravel()
108 mask = notna(xrav) & ymask.ravel()
109  
110 # See GH#5284, GH#5035, GH#19448 for historical reference
111 if mask.any():
112 result[mask] = op(xrav[mask], yrav[mask])
113  
114 else:
115 if not is_scalar(y):
116 raise TypeError(
117 f"Cannot broadcast np.ndarray with operand of type { type(y) }"
118 )
119  
120 # mask is only meaningful for x
121 result = np.empty(x.size, dtype=x.dtype)
122 mask = notna(xrav)
123  
124 # 1 ** np.nan is 1. So we have to unmask those.
125 if op is pow:
126 mask = np.where(x == 1, False, mask)
127 elif op is roperator.rpow:
128 mask = np.where(y == 1, False, mask)
129  
130 if mask.any():
131 result[mask] = op(xrav[mask], y)
132  
133 np.putmask(result, ~mask, np.nan)
134 result = result.reshape(x.shape) # 2D compat
135 return result
