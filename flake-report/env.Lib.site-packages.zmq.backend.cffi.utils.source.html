15 def has(capability):
16 """Check for zmq capability by name (e.g. 'ipc', 'curve')
17  
18 .. versionadded:: libzmq-4.1
19 .. versionadded:: 14.1
20 """
21 _check_version((4, 1), 'zmq.has')
22 if isinstance(capability, unicode):
23 capability = capability.encode('utf8')

27 def curve_keypair():
28 """generate a Z85 keypair for use with zmq.CURVE security
29  
30 Requires libzmq (≥ 4.0) to have been built with CURVE support.
31  
32 Returns
33 -------
34 (public, secret) : two bytestrings
35 The public and private keypair as 40 byte z85-encoded bytestrings.
36 """
37 _check_version((3, 2), "curve_keypair")
38 public = ffi.new('char[64]')
39 private = ffi.new('char[64]')
40 rc = C.zmq_curve_keypair(public, private)
41 _check_rc(rc)
42 return ffi.buffer(public)[:40], ffi.buffer(private)[:40]
