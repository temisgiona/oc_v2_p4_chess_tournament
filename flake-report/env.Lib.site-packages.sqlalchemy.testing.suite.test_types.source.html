50 def _literal_round_trip(self, type_, input_, output, filter_=None):
51 """test literal rendering """
52  
53 # for literal, we test the literal render in an INSERT
54 # into a typed column. we can then SELECT it back as its
55 # official type; ideally we'd be able to use CAST here
56 # but MySQL in particular can't CAST fully
57 t = Table("t", self.metadata, Column("x", type_))
58 t.create()
59  
60 with testing.db.connect() as conn:
61 for value in input_:
62 ins = (
63 t.insert()
64 .values(x=literal(value))
65 .compile(
66 dialect=testing.db.dialect,
67 compile_kwargs=dict(literal_binds=True),
68 )
69 )
70 conn.execute(ins)
71  
72 if self.supports_whereclause:
73 stmt = t.select().where(t.c.x == literal(value))
74 else:
75 stmt = t.select()
76  
77 stmt = stmt.compile(
78 dialect=testing.db.dialect,
79 compile_kwargs=dict(literal_binds=True),
80 )
81 for row in conn.execute(stmt):
82 value = row[0]
83 if filter_ is not None:
84 value = filter_(value)
85 assert value in output
