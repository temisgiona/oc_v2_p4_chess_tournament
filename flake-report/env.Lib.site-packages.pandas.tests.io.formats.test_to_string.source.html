192 def test_to_string_with_formatters():
193 df = DataFrame(
194 {
195 "int": [1, 2, 3],
196 "float": [1.0, 2.0, 3.0],
197 "object": [(1, 2), True, False],
198 },
199 columns=["int", "float", "object"],
200 )
201  
202 formatters = [
203 ("int", lambda x: f"0x{x:x}"),
204 ("float", lambda x: f"[{x: 4.1f}]"),
205 ("object", lambda x: f"-{x!s}-"),
206 ]
207 result = df.to_string(formatters=dict(formatters))
208 result2 = df.to_string(formatters=list(zip(*formatters))[1])
209 assert result == (
210 " int float object\n"
211 "0 0x1 [ 1.0] -(1, 2)-\n"
212 "1 0x2 [ 2.0] -True-\n"
213 "2 0x3 [ 3.0] -False-"
214 )
215 assert result == result2
