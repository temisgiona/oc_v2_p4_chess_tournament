42 def _wire_kernel(self):
43 """Initializes the kernel inside GTK.
44  
45 This is meant to run only once at startup, so it does its job and
46 returns False to ensure it doesn't get run again by GTK.
47 """
48 self.gtk_main, self.gtk_main_quit = self._hijack_gtk()
