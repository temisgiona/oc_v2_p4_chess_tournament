18 def cast_bytes(s, encoding='utf8', errors='strict'):
19 """cast unicode or bytes to bytes"""
20 if isinstance(s, bytes):
21 return s
22 elif isinstance(s, unicode):
23 return s.encode(encoding, errors)
24 else:
25 raise TypeError("Expected unicode or bytes, got %r" % s)

28 def cast_unicode(s, encoding='utf8', errors='strict'):
29 """cast bytes or unicode to unicode"""

32 elif isinstance(s, unicode):
