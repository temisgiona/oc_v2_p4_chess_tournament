103 def propagate_after_start(self, event):
104 # pydevd starts sending events as soon as we connect, but the client doesn't
105 # expect to see any until it receives the response to "launch" or "attach"
106 # request. If client is not ready yet, save the event instead of propagating
107 # it immediately.
108 if self._deferred_events is not None:
109 self._deferred_events.append(event)
110 log.debug("Propagation deferred.")
111 else:
112 self.client.channel.propagate(event)

114 def _propagate_deferred_events(self):
115 log.debug("Propagating deferred events to {0}...", self.client)
116 for event in self._deferred_events:
117 log.debug("Propagating deferred {0}", event.describe())
118 self.client.channel.propagate(event)
119 log.info("All deferred events propagated to {0}.", self.client)
120 self._deferred_events = None

199 request("debugOptions", json.array(unicode))
200 )

202 f(self, request)
203 if request.response is not None:
204 return
205  
206 if self.server:
207 self.server.initialize(self._initialize_request)
208 self._initialize_request = None

210 arguments = request.arguments
211 if self.launcher:
212 if "RedirectOutput" in debug_options:
213 # The launcher is doing output redirection, so we don't need the
214 # server to do it, as well.
215 arguments = dict(arguments)
216 arguments["debugOptions"] = list(
217 debug_options - {"RedirectOutput"}
218 )

220 if arguments.get("redirectOutput"):
221 arguments = dict(arguments)

224 # pydevd doesn't send "initialized", and responds to the start request
225 # immediately, without waiting for "configurationDone". If it changes
226 # to conform to the DAP spec, we'll need to defer waiting for response.
227 try:
228 self.server.channel.request(request.command, arguments)
229 except messaging.NoMoreMessages:
230 # Server closed connection before we could receive the response to
231 # "attach" or "launch" - this can happen when debuggee exits shortly
232 # after starting. It's not an error, but we can't do anything useful
233 # here at this point, either, so just bail out.
234 request.respond({})
235 self.session.finalize(
236 fmt(
237 "{0} disconnected before responding to {1!j}",
238 self.server,
239 request.command,
240 )
241 )
242 return
243 except messaging.MessageHandlingError as exc:
244 exc.propagate(request)
245  
246 if self.session.no_debug:
247 self.start_request = request
248 self.has_started = True
249 request.respond({})
250 self._propagate_deferred_events()
251 return

446 # There are four distinct possibilities here.
447 #
448 # If "processId" is specified, this is attach-by-PID. We need to inject the
449 # debug server into the designated process, and then wait until it connects
450 # back to us. Since the injected server can crash, there must be a timeout.
451 #
452 # If "subProcessId" is specified, this is attach to a known subprocess, likely
453 # in response to a "debugpyAttach" event. If so, the debug server should be
454 # connected already, and thus the wait timeout is zero.
455 #
456 # If "listen" is specified, this is attach-by-socket with the server expected
457 # to connect to the adapter via debugpy.connect(). There is no PID known in
458 # advance, so just wait until the first server connection indefinitely, with
459 # no timeout.
460 #
461 # If "connect" is specified, this is attach-by-socket in which the server has
462 # spawned the adapter via debugpy.listen(). There is no PID known to the client
463 # in advance, but the server connection should be either be there already, or
464 # the server should be connecting shortly, so there must be a timeout.
465 #
466 # In the last two cases, if there's more than one server connection already,
467 # this is a multiprocess re-attach. The client doesn't know the PID, so we just
468 # connect it to the oldest server connection that we have - in most cases, it
469 # will be the one for the root debuggee process, but if it has exited already,
470 # it will be some subprocess.
471 if pid != ():
472 if not isinstance(pid, int):
473 try:
474 pid = int(pid)
475 except Exception:
476 raise request.isnt_valid('"processId" must be parseable as int')
477 debugpy_args = request("debugpyArgs", json.array(unicode))
478 servers.inject(pid, debugpy_args)
479 timeout = common.PROCESS_SPAWN_TIMEOUT

480 pred = lambda conn: conn.pid == pid
481 else:
482 if sub_pid == ():
  • E731 Do not assign a lambda expression, use a def
483 pred = lambda conn: True
484 timeout = common.PROCESS_SPAWN_TIMEOUT if listen == () else None
485 else:
  • E731 Do not assign a lambda expression, use a def
486 pred = lambda conn: conn.pid == sub_pid
487 timeout = 0
488  
489 self.channel.send_event("debugpyWaitingForServer", {"host": host, "port": port})
490 conn = servers.wait_for_connection(self.session, pred, timeout)
491 if conn is None:
492 if sub_pid != ():
493 # If we can't find a matching subprocess, it's not always an error -
494 # it might have already exited, or didn't even get a chance to connect.
495 # To prevent the client from complaining, pretend that the "attach"
496 # request was successful, but that the session terminated immediately.
497 request.respond({})
498 self.session.finalize(
499 fmt('No known subprocess with "subProcessId":{0}', sub_pid)
500 )
501 return

513 conn.attach_to_session(self.session)

614 def notify_of_subprocess(self, conn):
615 with self.session:
616 if self.start_request is None or conn in self._known_subprocesses:
617 return
618 if "processId" in self.start_request.arguments:
619 log.warning(
620 "Not reporting subprocess for {0}, because the parent process "
621 'was attached to using "processId" rather than "port".',
622 self.session,
623 )
624 return
625  
626 log.info("Notifying {0} about {1}.", self, conn)
627 body = dict(self.start_request.arguments)
628 self._known_subprocesses.add(conn)
629  
630 for key in "processId", "listen", "preLaunchTask", "postDebugTask":
631 body.pop(key, None)
632  
633 body["name"] = fmt("Subprocess {0}", conn.pid)
634 body["request"] = "attach"
635 body["subProcessId"] = conn.pid
636  
637 for key in "args", "processName", "pythonArgs":
638 body.pop(key, None)
639  
640 host = body.pop("host", None)
641 port = body.pop("port", None)
642 if "connect" not in body:
643 body["connect"] = {}
644 if "host" not in body["connect"]:
645 body["connect"]["host"] = host if host is not None else "127.0.0.1"
646 if "port" not in body["connect"]:
647 if port is None:
648 _, port = listener.getsockname()
649 body["connect"]["port"] = port
650  
651 self.channel.send_event("debugpyAttach", body)
