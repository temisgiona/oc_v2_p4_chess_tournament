83 def _excel2num(x: str) -> int:
84 """
85 Convert Excel column name like 'AB' to 0-based column index.
86  
87 Parameters
88 ----------
89 x : str
90 The Excel column name to convert to a 0-based column index.
91  
92 Returns
93 -------
94 num : int
95 The column index corresponding to the name.
96  
97 Raises
98 ------
99 ValueError
100 Part of the Excel column name was invalid.
101 """
102 index = 0
103  
104 for c in x.upper().strip():
105 cp = ord(c)
106  
107 if cp < ord("A") or cp > ord("Z"):
108 raise ValueError(f"Invalid column name: {x}")
109  
110 index = index * 26 + cp - ord("A") + 1
111  
112 return index - 1

115 def _range2cols(areas: str) -> list[int]:
116 """
117 Convert comma separated list of column names and ranges to indices.
118  
119 Parameters
120 ----------
121 areas : str
122 A string containing a sequence of column ranges (or areas).
123  
124 Returns
125 -------
126 cols : list
127 A list of 0-based column indices.
128  
129 Examples
130 --------
131 >>> _range2cols('A:E')
132 [0, 1, 2, 3, 4]
133 >>> _range2cols('A,C,Z:AB')
134 [0, 2, 25, 26, 27]
135 """
136 cols: list[int] = []
137  
138 for rng in areas.split(","):
139 if ":" in rng:
140 rngs = rng.split(":")
141 cols.extend(range(_excel2num(rngs[0]), _excel2num(rngs[1]) + 1))
142 else:
143 cols.append(_excel2num(rng))
144  
145 return cols
