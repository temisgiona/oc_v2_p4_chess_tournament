118 def _getcallname(self, instr):
119 if instr.name == 'BINARY_SUBSCR':
120 return '__getitem__().__call__'

197 def _handle_call_from_instr(self, func_name_instr, func_call_instr):
198 self.load_attrs.pop(_TargetIdHashable(func_name_instr), None)
199 call_name = self._getcallname(func_name_instr)
200 target = None
201 if not call_name:
202 pass # Ignore if we can't identify a name
203 elif call_name in ('<listcomp>', '<genexpr>', '<setcomp>', '<dictcomp>'):
204 code_obj = self.func_name_id_to_code_object[_TargetIdHashable(func_name_instr)]
205 if code_obj is not None:
206 children_targets = _get_smart_step_into_targets(code_obj)
207 if children_targets:
208 # i.e.: we have targets inside of a <listcomp> or <genexpr>.
209 # Note that to actually match this in the debugger we need to do matches on 2 frames,
210 # the one with the <listcomp> and then the actual target inside the <listcomp>.
211 target = Target(call_name, func_name_instr.lineno, func_call_instr.offset, children_targets)
212 self.function_calls.append(
213 target)
214  
215 else:
216 # Ok, regular call
217 target = Target(call_name, func_name_instr.lineno, func_call_instr.offset)
218 self.function_calls.append(target)
219  
220 if DEBUG and target is not None:
221 print('Created target', target)
222 self._stack.append(func_call_instr) # Keep the func call as the result

653 def _get_smart_step_into_targets(code):
654 '''
655 :return list(Target)
656 '''
657 b = bytecode.Bytecode.from_code(code)
658 cfg = bytecode_cfg.ControlFlowGraph.from_bytecode(b)
659  
660 ret = []
661  
662 for block in cfg:
663 if DEBUG:
664 print('\nStart block----')
665 stack = _StackInterpreter(block)
666 for instr in block:
667 try:
668 func_name = 'on_%s' % (instr.name,)
669 func = getattr(stack, func_name, None)
670 if func is None:
671 if STRICT_MODE:
672 raise AssertionError('%s not found.' % (func_name,))
673 else:
674 continue
675 if DEBUG:
676 print('\nWill handle: ', instr, '>>', stack._getname(instr), '<<')
677 func(instr)
