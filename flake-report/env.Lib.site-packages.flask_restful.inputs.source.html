75 """Normalize datetime intervals.
76  
77 Given a pair of datetime.date or datetime.datetime objects,
78 returns a 2-tuple of tz-aware UTC datetimes spanning the same interval.
79  
80 For datetime.date objects, the returned interval starts at 00:00:00.0
81 on the first date and ends at 00:00:00.0 on the second.
82  
83 Naive datetimes are upgraded to UTC.
84  
85 Timezone-aware datetimes are normalized to the UTC tzdata.
86  
87 Params:
88 - start: A date or datetime
89 - end: A date or datetime

91 if not isinstance(start, datetime):
92 start = datetime.combine(start, START_OF_DAY)

97 end = pytz.UTC.localize(end)

99 start = start.astimezone(pytz.UTC)
100 end = end.astimezone(pytz.UTC)

105 def _expand_datetime(start, value):
106 if not isinstance(start, datetime):
107 # Expand a single date object to be the interval spanning
108 # that entire day.
109 end = start + timedelta(days=1)
110 else:
111 # Expand a datetime based on the finest resolution provided
112 # in the original input string.
113 time = value.split('T')[1]
114 time_without_offset = re.sub('[+-].+', '', time)
115 num_separators = time_without_offset.count(':')
116 if num_separators == 0:
117 # Hour resolution
118 end = start + timedelta(hours=1)
119 elif num_separators == 1:
120 # Minute resolution:
121 end = start + timedelta(minutes=1)
122 else:
123 # Second resolution
124 end = start + timedelta(seconds=1)
125  
126 return end

142 def iso8601interval(value, argument='argument'):
143 """Parses ISO 8601-formatted datetime intervals into tuples of datetimes.
144  
145 Accepts both a single date(time) or a full interval using either start/end
146 or start/duration notation, with the following behavior:
147  
148 - Intervals are defined as inclusive start, exclusive end
149 - Single datetimes are translated into the interval spanning the
150 largest resolution not specified in the input value, up to the day.
151 - The smallest accepted resolution is 1 second.
152 - All timezones are accepted as values; returned datetimes are
153 localized to UTC. Naive inputs and date inputs will are assumed UTC.
154  
155 Examples::
156  
157 "2013-01-01" -> datetime(2013, 1, 1), datetime(2013, 1, 2)
158 "2013-01-01T12" -> datetime(2013, 1, 1, 12), datetime(2013, 1, 1, 13)
159 "2013-01-01/2013-02-28" -> datetime(2013, 1, 1), datetime(2013, 2, 28)
160 "2013-01-01/P3D" -> datetime(2013, 1, 1), datetime(2013, 1, 4)
161 "2013-01-01T12:00/PT30M" -> datetime(2013, 1, 1, 12), datetime(2013, 1, 1, 12, 30)
162 "2013-01-01T06:00/2013-01-01T12:00" -> datetime(2013, 1, 1, 6), datetime(2013, 1, 1, 12)
163  
164 :param str value: The ISO8601 date time as a string
165 :return: Two UTC datetimes, the start and the end of the specified interval
166 :rtype: A tuple (datetime, datetime)
167 :raises: ValueError, if the interval is invalid.
168 """
169  
170 try:
171 start, end = _parse_interval(value)
172  
173 if end is None:
174 end = _expand_datetime(start, value)
175  
176 start, end = _normalize_interval(start, end, value)
177  
178 except ValueError:
179 raise ValueError(
180 "Invalid {arg}: {value}. {arg} must be a valid ISO8601 "
181 "date/time interval.".format(arg=argument, value=value),
182 )
183  
184 return start, end
