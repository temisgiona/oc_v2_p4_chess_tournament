363 def poll_or_sleep_to_kernel_gone():
364 """
365 Poll until the kernel is not responding,
366 then wait (the subprocess), until process gone.
367  
368 After this function the kernel is either:
369 - still responding; or
370 - subprocess has been culled.
371 """
372 if self.is_alive():
373 time.sleep(pollinterval)
374 else:
375 # If there's still a proc, wait and clear
376 if self.has_kernel:
377 self.kernel.wait()
378 self.kernel = None
379 return True
380  
381 # wait 50% of the shutdown timeout...
382 for i in range(int(waittime / 2 / pollinterval)):
383 if poll_or_sleep_to_kernel_gone():
384 break
385 else:
386 # if we've exited the loop normally (no break)
387 # send sigterm and wait the other 50%.
388 self.log.debug("Kernel is taking too long to finish, terminating")
389 self._shutdown_status = _ShutdownStatus.SigtermRequest
390 self._send_kernel_sigterm()
391 for i in range(int(waittime / 2 / pollinterval)):
392 if poll_or_sleep_to_kernel_gone():
393 break
394 else:
395 # OK, we've waited long enough.
396 if self.has_kernel:
397 self.log.debug("Kernel is taking too long to finish, killing")
398 self._shutdown_status = _ShutdownStatus.SigkillRequest
399 self._kill_kernel()

419 def shutdown_kernel(self, now=False, restart=False):
420 """Attempts to stop the kernel process cleanly.
421  
422 This attempts to shutdown the kernels cleanly by:
423  
424 1. Sending it a shutdown message over the control channel.
425 2. If that fails, the kernel is shutdown forcibly by sending it
426 a signal.
427  
428 Parameters
429 ----------
430 now : bool
431 Should the kernel be forcible killed *now*. This skips the
432 first, nice shutdown attempt.
433 restart: bool
434 Will this kernel be restarted after it is shutdown. When this
435 is True, connection files will not be cleaned up.
436 """
437 self.shutting_down = True # Used by restarter to prevent race condition
438 # Stop monitoring for restarting while we shutdown.
439 self.stop_restarter()
440  
441 self.interrupt_kernel()
442  
443 if now:
444 self._kill_kernel()
445 else:
446 self.request_shutdown(restart=restart)
447 # Don't send any additional kernel kill messages immediately, to give
448 # the kernel a chance to properly execute shutdown actions. Wait for at
449 # most 1s, checking every 0.1s.
450 self.finish_shutdown()
451  
452 # In 6.1.5, a new method, cleanup_resources(), was introduced to address
453 # a leak issue (https://github.com/jupyter/jupyter_client/pull/548) and
454 # replaced the existing cleanup() method. However, that method introduction
455 # breaks subclass implementations that override cleanup() since it would
456 # circumvent cleanup() functionality implemented in subclasses.
457 # By detecting if the current instance overrides cleanup(), we can determine
458 # if the deprecated path of calling cleanup() should be performed - which avoids
459 # unnecessary deprecation warnings in a majority of configurations in which
460 # subclassed KernelManager instances are not in use.
461 # Note: because subclasses may have already implemented cleanup_resources()
462 # but need to support older jupyter_clients, we should only take the deprecated
463 # path if cleanup() is overridden but cleanup_resources() is not.
464  
465 overrides_cleanup = type(self).cleanup is not KernelManager.cleanup
466 overrides_cleanup_resources = type(self).cleanup_resources is not KernelManager.cleanup_resources
467  
468 if overrides_cleanup and not overrides_cleanup_resources:
469 self.cleanup(connection_file=not restart)
470 else:
471 self.cleanup_resources(restart=restart)

548 """Kill the running kernel.

551 """

554 # TerminateProcess() on Win32).

559 self.kernel.kill()

725 # the kernel a chance to properly execute shutdown actions. Wait for at
726 # most 1s, checking every 0.1s.
727 await self.finish_shutdown()
728  
729 # See comment in KernelManager.shutdown_kernel().
730 overrides_cleanup = type(self).cleanup is not AsyncKernelManager.cleanup
731 overrides_cleanup_resources = type(self).cleanup_resources is not AsyncKernelManager.cleanup_resources
732  
733 if overrides_cleanup and not overrides_cleanup_resources:
734 self.cleanup(connection_file=not restart)
735 else:
736 self.cleanup_resources(restart=restart)

738 async def restart_kernel(self, now=False, newports=False, **kw):
739 """Restarts a kernel with the arguments that were used to launch it.
740  
741 Parameters
742 ----------
743 now : bool, optional
744 If True, the kernel is forcefully restarted *immediately*, without
745 having a chance to do any cleanup action. Otherwise the kernel is
746 given 1s to clean up before a forceful restart is issued.
747  
748 In all cases the kernel is restarted, the only difference is whether
749 it is given a chance to perform a clean shutdown or not.
750  
751 newports : bool, optional
752 If the old kernel was launched with random ports, this flag decides
753 whether the same ports and connection file will be used again.
754 If False, the same ports and connection file are used. This is
755 the default. If True, new random port numbers are chosen and a
756 new connection file is written. It is still possible that the newly
757 chosen random port numbers happen to be the same as the old ones.
758  
759 `**kw` : optional
760 Any options specified here will overwrite those used to launch the
761 kernel.
762 """
763 if self._launch_args is None:
764 raise RuntimeError("Cannot restart the kernel. "
765 "No previous call to 'start_kernel'.")
766 else:
767 # Stop currently running kernel.
768 await self.shutdown_kernel(now=now, restart=True)
769  
770 if newports:
771 self.cleanup_random_ports()
772  
773 # Start new kernel.
774 self._launch_args.update(kw)
775 await self.start_kernel(**self._launch_args)
776 return None

778 async def _send_kernel_sigterm(self):
779 """similar to _kill_kernel, but with sigterm (not sigkill), but do not block"""
780 if self.has_kernel:
781 # Signal the kernel to terminate (sends SIGTERM on Unix and
782 # if the kernel is a subprocess and we are on windows; this is
783 # equivalent to kill
784 try:
785 if hasattr(self.kernel, "terminate"):
786 self.kernel.terminate()
787 elif hasattr(signal, "SIGTERM"):
788 await self.signal_kernel(signal.SIGTERM)
789 else:
790 self.log.debug(
791 "Cannot set term signal to kernel, no"
792 " `.terminate()` method and no values for SIGTERM"
793 )
794 except OSError as e:
795 # In Windows, we will get an Access Denied error if the process
796 # has already terminated. Ignore it.
797 if sys.platform == "win32":
798 if e.winerror != 5:
799 raise
800 # On Unix, we may get an ESRCH error if the process has already
801 # terminated. Ignore it.
802 else:
803 from errno import ESRCH
804  
805 if e.errno != ESRCH:
806 raise

808 async def _kill_kernel(self):
809 """Kill the running kernel.
810  
811 This is a private method, callers should use shutdown_kernel(now=True).
812 """
813 if self.has_kernel:
814 # Signal the kernel to terminate (sends SIGKILL on Unix and calls
815 # TerminateProcess() on Win32).
816 try:
817 if hasattr(signal, 'SIGKILL'):
818 await self.signal_kernel(signal.SIGKILL)
819 else:
820 self.kernel.kill()
821 except OSError as e:
822 # In Windows, we will get an Access Denied error if the process
823 # has already terminated. Ignore it.
824 if sys.platform == 'win32':
825 if e.winerror != 5:
826 raise
827 # On Unix, we may get an ESRCH error if the process has already
828 # terminated. Ignore it.
829 else:
830 from errno import ESRCH
831 if e.errno != ESRCH:
832 raise
833  
834 # Wait until the kernel terminates.
835 try:
836 await asyncio.wait_for(self._async_wait(), timeout=5.0)
837 except asyncio.TimeoutError:
838 # Wait timed out, just log warning but continue - not much more we can do.
839 self.log.warning("Wait for final termination of kernel timed out - continuing...")
840 pass
841 else:
842 # Process is no longer alive, wait and clear
843 if self.kernel is not None:
844 self.kernel.wait()
845 self.kernel = None

847 async def interrupt_kernel(self):
848 """Interrupts the kernel by sending it a signal.
849  
850 Unlike ``signal_kernel``, this operation is well supported on all
851 platforms.
852 """
853 if self.has_kernel:
854 interrupt_mode = self.kernel_spec.interrupt_mode
855 if interrupt_mode == 'signal':
856 if sys.platform == 'win32':
857 from .win_interrupt import send_interrupt
858 send_interrupt(self.kernel.win32_interrupt_event)
859 else:
860 await self.signal_kernel(signal.SIGINT)
861  
862 elif interrupt_mode == 'message':
863 msg = self.session.msg("interrupt_request", content={})
864 self._connect_control_socket()
865 self.session.send(self._control_socket, msg)
866 else:
867 raise RuntimeError("Cannot interrupt kernel. No kernel is running!")

869 async def signal_kernel(self, signum):
870 """Sends a signal to the process group of the kernel (this
871 usually includes the kernel and any subprocesses spawned by
872 the kernel).
873  
874 Note that since only SIGTERM is supported on Windows, this function is
875 only useful on Unix systems.
876 """
877 if self.has_kernel:
878 if hasattr(os, "getpgid") and hasattr(os, "killpg"):
879 try:
880 pgid = os.getpgid(self.kernel.pid)
881 os.killpg(pgid, signum)
882 return
883 except OSError:
884 pass
885 self.kernel.send_signal(signum)
886 else:
887 raise RuntimeError("Cannot signal kernel. No kernel is running!")

889 async def is_alive(self):
890 """Is the kernel process still running?"""
891 if self.has_kernel:
892 if self.kernel.poll() is None:
893 return True
894 else:
895 return False
896 else:
897 # we don't have a kernel
898 return False

900 async def _async_wait(self, pollinterval=0.1):
901 # Use busy loop at 100ms intervals, polling until the process is
902 # not alive. If we find the process is no longer alive, complete
903 # its cleanup via the blocking wait(). Callers are responsible for
904 # issuing calls to wait() using a timeout (see _kill_kernel()).
905 while await self.is_alive():
906 await asyncio.sleep(pollinterval)

913 """Start a new kernel, and return its Manager and Client"""
914 km = KernelManager(kernel_name=kernel_name)

917 kc.start_channels()
918 try:
919 kc.wait_for_ready(timeout=startup_timeout)
920 except RuntimeError:
921 kc.stop_channels()
922 km.shutdown_kernel()
923 raise
924
