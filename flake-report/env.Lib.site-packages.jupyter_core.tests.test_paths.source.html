274 def fetch_win32_permissions(filename):
275 '''Extracts file permissions on windows using icacls'''
276 role_permissions = {}
277 for index, line in enumerate(os.popen("icacls %s" % filename).read().splitlines()):
278 if index == 0:
279 line = line.split(filename)[-1].strip().lower()
280 match = re.match(r'\s*([^:]+):\(([^\)]*)\)', line)
281 if match:
282 usergroup, permissions = match.groups()
283 usergroup = usergroup.lower().split('\\')[-1]
284 permissions = set(p.lower() for p in permissions.split(','))
285 role_permissions[usergroup] = permissions
286 elif not line.strip():
287 break
288 return role_permissions
