53 def get_version(module: types.ModuleType) -> str:
54 version = getattr(module, "__version__", None)
55 if version is None:
56 # xlrd uses a capitalized attribute name
57 version = getattr(module, "__VERSION__", None)
58  
59 if version is None:
60 raise ImportError(f"Can't determine version for {module.__name__}")
61 return version
