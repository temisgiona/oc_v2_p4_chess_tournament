139 # Determines whether an iterable exceeds the limits set in
140 # maxlimits, and is therefore unsafe to repr().
141 def _is_long_iter(self, obj, level=0):
142 try:
143 # Strings have their own limits (and do not nest). Because
144 # they don't have __iter__ in 2.x, this check goes before
145 # the next one.
146 if isinstance(obj, self.string_types):
147 return len(obj) > self.maxstring_inner
148  
149 # If it's not an iterable (and not a string), it's fine.
150 if not hasattr(obj, '__iter__'):
151 return False
152  
153 # If it's not an instance of these collection types then it
154 # is fine. Note: this is a fix for
155 # https://github.com/Microsoft/ptvsd/issues/406
156 if not isinstance(obj, self.long_iter_types):
157 return False
158  
159 # Iterable is its own iterator - this is a one-off iterable
160 # like generator or enumerate(). We can't really count that,
161 # but repr() for these should not include any elements anyway,
162 # so we can treat it the same as non-iterables.
163 if obj is iter(obj):
164 return False
165  
166 # xrange reprs fine regardless of length.
167 if isinstance(obj, xrange):
168 return False
169  
170 # numpy and scipy collections (ndarray etc) have
171 # self-truncating repr, so they're always safe.
172 try:
173 module = type(obj).__module__.partition('.')[0]
174 if module in ('numpy', 'scipy'):
175 return False
176 except Exception:
177 pass
178  
179 # Iterables that nest too deep are considered long.
180 if level >= len(self.maxcollection):
181 return True
182  
183 # It is too long if the length exceeds the limit, or any
184 # of its elements are long iterables.
185 if hasattr(obj, '__len__'):
186 try:
187 size = len(obj)
188 except Exception:
189 size = None
190 if size is not None and size > self.maxcollection[level]:
191 return True
192 return any((self._is_long_iter(item, level + 1) for item in obj)) # noqa
193 return any(i > self.maxcollection[level] or self._is_long_iter(item, level + 1) for i, item in enumerate(obj)) # noqa
194  
195 except Exception:
196 # If anything breaks, assume the worst case.
197 return True
