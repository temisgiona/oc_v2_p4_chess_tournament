16 # Z85CHARS is the base 85 symbol table
17 Z85CHARS = b"0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ.-:+=^!/*?&<>()[]{}@%$#"
18 # Z85MAP maps integers in [0,84] to the appropriate character in Z85CHARS
19 Z85MAP = dict([(c, idx) for idx, c in enumerate(Z85CHARS)])
20  
21 _85s = [85 ** i for i in range(5)][::-1]

24 def encode(rawbytes):
25 """encode raw bytes into Z85"""
26 # Accepts only byte arrays bounded to 4 bytes
27 if len(rawbytes) % 4:
28 raise ValueError("length must be multiple of 4, not %i" % len(rawbytes))
29  
30 nvalues = len(rawbytes) / 4
31  
32 values = struct.unpack('>%dI' % nvalues, rawbytes)
33 encoded = []
34 for v in values:
35 for offset in _85s:
36 encoded.append(Z85CHARS[(v // offset) % 85])
37  
38 return bytes(encoded)

41 def decode(z85bytes):
42 """decode Z85 bytes to raw bytes, accepts ASCII string"""
43 if isinstance(z85bytes, str):
44 try:
45 z85bytes = z85bytes.encode('ascii')
46 except UnicodeEncodeError:
47 raise ValueError('string argument should contain only ASCII characters')
48  
49 if len(z85bytes) % 5:
50 raise ValueError("Z85 length must be multiple of 5, not %i" % len(z85bytes))
51  
52 nvalues = len(z85bytes) / 5
53 values = []
54 for i in range(0, len(z85bytes), 5):
55 value = 0
56 for j, offset in enumerate(_85s):
57 value += Z85MAP[z85bytes[i + j]] * offset
58 values.append(value)
59 return struct.pack('>%dI' % nvalues, *values)
