30 def register(self, socket, flags: int = POLLIN | POLLOUT):
31 """p.register(socket, flags=POLLIN|POLLOUT)
32  
33 Register a 0MQ socket or native fd for I/O monitoring.
34  
35 register(s,0) is equivalent to unregister(s).
36  
37 Parameters
38 ----------
39 socket : zmq.Socket or native socket
40 A zmq.Socket or any Python object having a ``fileno()``
41 method that returns a valid file descriptor.
42 flags : int
43 The events to watch for. Can be POLLIN, POLLOUT or POLLIN|POLLOUT.
44 If `flags=0`, socket will be unregistered.
45 """
46 if flags:
47 if socket in self._map:
48 idx = self._map[socket]
49 self.sockets[idx] = (socket, flags)
50 else:
51 idx = len(self.sockets)
52 self.sockets.append((socket, flags))
53 self._map[socket] = idx
54 elif socket in self._map:
55 # uregister sockets registered with no events
56 self.unregister(socket)
57 else:
58 # ignore new sockets with no events
59 pass

61 def modify(self, socket, flags=POLLIN | POLLOUT):
62 """Modify the flags for an already registered 0MQ socket or native fd."""
63 self.register(socket, flags)
64  
65 def unregister(self, socket):
66 """Remove a 0MQ socket or native fd for I/O monitoring.
67  
68 Parameters
69 ----------
70 socket : Socket
71 The socket instance to stop polling.
72 """
73 idx = self._map.pop(socket)
74 self.sockets.pop(idx)
75 # shift indices after deletion
76 for socket, flags in self.sockets[idx:]:
77 self._map[socket] -= 1

79 def poll(self, timeout: Optional[int] = None):
80 """Poll the registered 0MQ or native fds for I/O.
81  
82 If there are currently events ready to be processed, this function will return immediately.
83 Otherwise, this function will return as soon the first event is available or after timeout
84 milliseconds have elapsed.
85  
86 Parameters
87 ----------
88 timeout : float, int
89 The timeout in milliseconds. If None, no `timeout` (infinite). This
90 is in milliseconds to be compatible with ``select.poll()``.
91  
92 Returns
93 -------
94 events : list of tuples
95 The list of events that are ready to be processed.
96 This is a list of tuples of the form ``(socket, event_mask)``, where the 0MQ Socket
97 or integer fd is the first element, and the poll event mask (POLLIN, POLLOUT) is the second.
98 It is common to call ``events = dict(poller.poll())``,
99 which turns the list of tuples into a mapping of ``socket : event_mask``.
100 """
101 if timeout is None or timeout < 0:
102 timeout = -1
103 elif isinstance(timeout, float):
104 timeout = int(timeout)
105 return zmq_poll(self.sockets, timeout=timeout)

108 def select(rlist: List, wlist: List, xlist: List, timeout: Optional[int] = None):
109 """select(rlist, wlist, xlist, timeout=None) -> (rlist, wlist, xlist)
110  
111 Return the result of poll as a lists of sockets ready for r/w/exception.
112  
113 This has the same interface as Python's built-in ``select.select()`` function.
114  
115 Parameters
116 ----------
117 timeout : float, int, optional
118 The timeout in seconds. If None, no timeout (infinite). This is in seconds to be
119 compatible with ``select.select()``.
120 rlist : list of sockets/FDs
121 sockets/FDs to be polled for read events
122 wlist : list of sockets/FDs
123 sockets/FDs to be polled for write events
124 xlist : list of sockets/FDs
125 sockets/FDs to be polled for error events
126  
127 Returns
128 -------
129 (rlist, wlist, xlist) : tuple of lists of sockets (length 3)
130 Lists correspond to sockets available for read/write/error events respectively.
131 """
132 if timeout is None:
133 timeout = -1
134 # Convert from sec -> us for zmq_poll.
135 # zmq_poll accepts 3.x style timeout in ms
136 timeout = int(timeout * 1000.0)
137 if timeout < 0:
138 timeout = -1
139 sockets = []
140 for s in set(rlist + wlist + xlist):
141 flags = 0
142 if s in rlist:
143 flags |= POLLIN
144 if s in wlist:
145 flags |= POLLOUT
146 if s in xlist:
147 flags |= POLLERR
148 sockets.append((s, flags))
149 return_sockets = zmq_poll(sockets, timeout)
150 rlist, wlist, xlist = [], [], []
151 for s, flags in return_sockets:
152 if flags & POLLIN:
153 rlist.append(s)
154 if flags & POLLOUT:
155 wlist.append(s)
156 if flags & POLLERR:
157 xlist.append(s)
158 return rlist, wlist, xlist
