18 class TestSocketMonitor(BaseZMQTestCase):
19 @require_zmq_4
20 def test_monitor(self):
21 """Test monitoring interface for sockets."""
22 s_rep = self.context.socket(zmq.REP)
23 s_req = self.context.socket(zmq.REQ)
24 self.sockets.extend([s_rep, s_req])
25 s_req.bind("tcp://127.0.0.1:6666")
26 # try monitoring the REP socket
27  
28 s_rep.monitor(
29 "inproc://monitor.rep",
30 zmq.EVENT_CONNECT_DELAYED | zmq.EVENT_CONNECTED | zmq.EVENT_MONITOR_STOPPED,
31 )
32 # create listening socket for monitor
33 s_event = self.context.socket(zmq.PAIR)
34 self.sockets.append(s_event)
35 s_event.connect("inproc://monitor.rep")
36 s_event.linger = 0
37 # test receive event for connect event
38 s_rep.connect("tcp://127.0.0.1:6666")
39 m = recv_monitor_message(s_event)
40 if m['event'] == zmq.EVENT_CONNECT_DELAYED:
41 self.assertEqual(m['endpoint'], b"tcp://127.0.0.1:6666")
42 # test receive event for connected event
43 m = recv_monitor_message(s_event)
44 self.assertEqual(m['event'], zmq.EVENT_CONNECTED)
45 self.assertEqual(m['endpoint'], b"tcp://127.0.0.1:6666")
46  
47 # test monitor can be disabled.
48 s_rep.disable_monitor()
49 m = recv_monitor_message(s_event)
50 self.assertEqual(m['event'], zmq.EVENT_MONITOR_STOPPED)
