320 def _get_reverse_ansi_sequences() -> Dict[Keys, str]:
321 """
322 Create a dictionary that maps prompt_toolkit keys back to the VT100 escape
323 sequences.
324 """
325 result: Dict[Keys, str] = {}
326  
327 for sequence, key in ANSI_SEQUENCES.items():
328 if not isinstance(key, tuple):
329 if key not in result:
330 result[key] = sequence
331  
332 return result

335 REVERSE_ANSI_SEQUENCES = _get_reverse_ansi_sequences()
