163 def check_round_trip(
164 df,
165 engine=None,
166 path=None,
167 write_kwargs=None,
168 read_kwargs=None,
169 expected=None,
170 check_names=True,
171 check_like=False,
172 check_dtype=True,
173 repeat=2,
174 ):
175 """Verify parquet serializer and deserializer produce the same results.
176  
177 Performs a pandas to disk and disk to pandas round trip,
178 then compares the 2 resulting DataFrames to verify equality.
179  
180 Parameters
181 ----------
182 df: Dataframe
183 engine: str, optional
184 'pyarrow' or 'fastparquet'
185 path: str, optional
186 write_kwargs: dict of str:str, optional
187 read_kwargs: dict of str:str, optional
188 expected: DataFrame, optional
189 Expected deserialization result, otherwise will be equal to `df`
190 check_names: list of str, optional
191 Closed set of column names to be compared
192 check_like: bool, optional
193 If True, ignore the order of index & columns.
194 repeat: int, optional
195 How many times to repeat the test
196 """
197 write_kwargs = write_kwargs or {"compression": None}
198 read_kwargs = read_kwargs or {}
199  
200 if expected is None:
201 expected = df
202  
203 if engine:
204 write_kwargs["engine"] = engine
205 read_kwargs["engine"] = engine
206  
207 def compare(repeat):
208 for _ in range(repeat):
209 df.to_parquet(path, **write_kwargs)
210 with catch_warnings(record=True):
211 actual = read_parquet(path, **read_kwargs)
212  
213 tm.assert_frame_equal(
214 expected,
215 actual,
216 check_names=check_names,
217 check_like=check_like,
218 check_dtype=check_dtype,
219 )
220  
221 if path is None:
222 with tm.ensure_clean() as path:
223 compare(repeat)
224 else:
225 compare(repeat)
