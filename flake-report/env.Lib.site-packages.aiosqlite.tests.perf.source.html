20 def timed(fn, name=None):
21 """
22 Decorator for perf testing a block of async code.
23  
24 Expects the wrapped function to return an async generator.
25 The generator should do setup, then yield when ready to start perf testing.
26 The decorator will then pump the generator repeatedly until the target
27 time has been reached, then close the generator and print perf results.
28 """
29  
30 name = name or fn.__name__
31  
32 async def wrapper(*args, **kwargs):
33 gen = fn(*args, **kwargs)
34  
35 await gen.asend(None)
36 count = 0
37 before = time.time()
38  
39 while True:
40 count += 1
41 value = time.time() - before < TARGET
42 try:
43 if value:
44 await gen.asend(value)
45 else:
46 await gen.aclose()
47 break
48  
49 except StopAsyncIteration:
50 break
51  
52 except Exception as e:
53 print(f"exception occurred: {e}")
54 return
55  
56 duration = time.time() - before
57  
58 RESULTS[name] = (count, duration)
59  
60 return wrapper
