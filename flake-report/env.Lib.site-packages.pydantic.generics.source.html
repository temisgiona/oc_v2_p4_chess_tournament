145 def replace_types(type_: Any, type_map: Mapping[Any, Any]) -> Any:
146 """Return type with all occurances of `type_map` keys recursively replaced with their values.
147  
148 :param type_: Any type, class or generic alias

220 def iter_contained_typevars(v: Any) -> Iterator[TypeVarType]:
221 """Recursively iterate through all subtypes and type args of `v` and yield any typevars that are found."""
222 if isinstance(v, TypeVar):
223 yield v
224 elif hasattr(v, '__parameters__') and not get_origin(v) and lenient_issubclass(v, GenericModel):
