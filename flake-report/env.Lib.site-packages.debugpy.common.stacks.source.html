19 def dump():
20 """Dump stacks of all threads in this process, except for the current thread.
21 """
22  
23 tid = threading.current_thread().ident
24 pid = os.getpid()
25  
26 log.info("Dumping stacks for process {0}...", pid)
27  
28 for t_ident, frame in sys._current_frames().items():
29 if t_ident == tid:
30 continue
31  
32 for t in threading.enumerate():
33 if t.ident == tid:
34 t_name = t.name
35 t_daemon = t.daemon
36 break
37 else:
38 t_name = t_daemon = "<unknown>"
39  
40 stack = "".join(traceback.format_stack(frame))
41 log.info(
42 "Stack of thread {0} (tid={1}, pid={2}, daemon={3}):\n\n{4}",
43 t_name,
44 t_ident,
45 pid,
46 t_daemon,
47 stack,
48 )
49  
50 log.info("Finished dumping stacks for process {0}.", pid)
