91 def _expand_classname(classname: str) -> List[str]:
92 """
93 Split a single class name at the `.` operator, and build a list of classes.

107 """
108 Take a style string, e.g. 'bg:red #88ff00 class:title'
109 and return a `Attrs` instance.
110 """

125 elif part == "italic":

127 elif part == "noitalic":
128 attrs = attrs._replace(italic=False)
129 elif part == "underline":
130 attrs = attrs._replace(underline=True)
131 elif part == "nounderline":
132 attrs = attrs._replace(underline=False)
133  
134 # prompt_toolkit extensions. Not in Pygments.
135 elif part == "blink":
136 attrs = attrs._replace(blink=True)
137 elif part == "noblink":
138 attrs = attrs._replace(blink=False)
139 elif part == "reverse":
140 attrs = attrs._replace(reverse=True)
141 elif part == "noreverse":
142 attrs = attrs._replace(reverse=False)
143 elif part == "hidden":
144 attrs = attrs._replace(hidden=True)
145 elif part == "nohidden":
146 attrs = attrs._replace(hidden=False)
147  
148 # Pygments properties that we ignore.
149 elif part in ("roman", "sans", "mono"):
150 pass
151 elif part.startswith("border:"):
152 pass
153

155 # Like '[transparent]' or '[set-cursor-position]'.
156 elif part.startswith("[") and part.endswith("]"):
157 pass
158

160 elif part.startswith("bg:"):
161 attrs = attrs._replace(bgcolor=parse_color(part[3:]))
162 elif part.startswith("fg:"): # The 'fg:' prefix is optional.

173 class Priority(Enum):
174 """
175 The priority of the rules, when a style is created from a dictionary.
176  
177 In a `Style`, rules that are defined later will always override previous
178 defined rules, however in a dictionary, the key order was arbitrary before
179 Python 3.6. This means that the style could change at random between rules.
180  
181 We have two options:
182  
183 - `DICT_KEY_ORDER`: This means, iterate through the dictionary, and take
184 the key/value pairs in order as they come. This is a good option if you
185 have Python >3.6. Rules at the end will override rules at the beginning.
186 - `MOST_PRECISE`: keys that are defined with most precision will get higher
187 priority. (More precise means: more elements.)
188 """
189  
190 DICT_KEY_ORDER = "KEY_ORDER"
191 MOST_PRECISE = "MOST_PRECISE"
192  
193  
194 # In the latest python verions, we take the dictionary ordering like it is,
195 # In older versions, we sort by by precision. If you need to write code that
196 # runs on all Python versions, it's best to sort them manually, with the most
197 # precise rules at the bottom.
198 if sys.version_info >= (3, 6):
199 default_priority = Priority.DICT_KEY_ORDER
200 else:
201 default_priority = Priority.MOST_PRECISE
