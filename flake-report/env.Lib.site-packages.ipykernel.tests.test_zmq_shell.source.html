17 class NoReturnDisplayHook(object):
18 """
19 A dummy DisplayHook which allows us to monitor
20 the number of times an object is called, but which
21 does *not* return a message when it is called.
22 """

29 class ReturnDisplayHook(NoReturnDisplayHook):
30 """
31 A dummy DisplayHook with the same counting ability
32 as its base class, but which also returns the same
33 message when it is called.
