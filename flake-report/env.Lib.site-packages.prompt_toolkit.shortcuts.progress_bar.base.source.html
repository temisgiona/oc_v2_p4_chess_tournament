337 def __iter__(self) -> Iterable[_CounterItem]:
338 if self.data is not None:
339 try:
340 for item in self.data:
341 yield item
342 self.item_completed()
343  
344 # Only done if we iterate to the very end.
345 self.done = True
346 finally:
347 # Ensure counter has stopped even if we did not iterate to the
348 # end (e.g. break or exceptions).
349 self.stopped = True

353 def item_completed(self) -> None:
354 """
355 Start handling the next item.
356  
357 (Can be called manually in case we don't have a collection to loop through.)

359 self.items_completed += 1

362 @property

364 """Whether a counter has been completed.
365  
366 Done counter have been stopped (see stopped) and removed depending on
367 remove_when_done value.

374 @done.setter

376 self._done = value
377 self.stopped = value

382 @property
383 def stopped(self) -> bool:
384 """Whether a counter has been stopped.
385  
386 Stopped counters no longer have increasing time_elapsed. This distinction is
387 also used to prevent the Bar formatter with unknown totals from continuing to run.
388  
389 A stopped counter (but not done) can be used to signal that a given counter has
390 encountered an error but allows other counters to continue
391 (e.g. download X of Y failed). Given how only done counters are removed
392 (see remove_when_done) this can help aggregate failures from a large number of
393 successes.
394  
395 Contrast this with done. A done counter has reached its 100% completion.
396 A stopped counter may be terminated before 100% completion.
397 """
398 return self.stop_time is not None
399  
400 @stopped.setter
401 def stopped(self, value: bool) -> None:
402 if value:
403 # This counter has not already been stopped.
404 if not self.stop_time:
405 self.stop_time = datetime.datetime.now()
406 else:
407 # Clearing any previously set stop_time.
408 self.stop_time = None

410 @property
411 def percentage(self) -> float:
412 if self.total is None:
413 return 0
414 else:
415 return self.items_completed * 100 / max(self.total, 1)

417 @property
418 def time_elapsed(self) -> datetime.timedelta:
419 """
420 Return how much time has been elapsed since the start.
421 """
422 if self.stop_time is None:
423 return datetime.datetime.now() - self.start_time
424 else:
425 return self.stop_time - self.start_time

427 @property
428 def time_left(self) -> Optional[datetime.timedelta]:
429 """
430 Timedelta representing the time left.
431 """
432 if self.total is None or not self.percentage:
433 return None
434 elif self.done or self.stopped:
435 return datetime.timedelta(0)
436 else:
437 return self.time_elapsed * (100 - self.percentage) / self.percentage
