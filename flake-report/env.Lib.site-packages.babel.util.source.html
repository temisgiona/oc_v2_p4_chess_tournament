25 def distinct(iterable):
26 """Yield all items in an iterable collection that are distinct.
27  
28 Unlike when using sets for a similar effect, the original ordering of the
29 items in the collection is preserved by this function.
30  
31 >>> print(list(distinct([1, 2, 1, 3, 4, 4])))
32 [1, 2, 3, 4]
33 >>> print(list(distinct('foobar')))
34 ['f', 'o', 'b', 'a', 'r']
35  
36 :param iterable: the iterable collection providing the data
37 """
38 seen = set()
39 for item in iter(iterable):
40 if item not in seen:
41 yield item
42 seen.add(item)

49 def parse_encoding(fp):
50 """Deduce the encoding of a source file from magic comment.
51  
52 It does this in the same way as the `Python interpreter`__
53  
  • E501 Line too long (92 > 79 characters)
54 .. __: https://docs.python.org/3.4/reference/lexical_analysis.html#encoding-declarations
55  
56 The ``fp`` argument should be a seekable file object.
57  
58 (From Jeff Dairiki)
59 """
60 pos = fp.tell()
61 fp.seek(0)
62 try:
63 line1 = fp.readline()
64 has_bom = line1.startswith(codecs.BOM_UTF8)
65 if has_bom:
66 line1 = line1[len(codecs.BOM_UTF8):]
67  
68 m = PYTHON_MAGIC_COMMENT_re.match(line1)
69 if not m:
70 try:
71 import ast
72 ast.parse(line1.decode('latin-1'))
73 except (ImportError, SyntaxError, UnicodeEncodeError):
74 # Either it's a real syntax error, in which case the source is
75 # not valid python source, or line2 is a continuation of line1,
76 # in which case we don't want to scan line2 for a magic
77 # comment.
78 pass
79 else:
80 line2 = fp.readline()
81 m = PYTHON_MAGIC_COMMENT_re.match(line2)
82  
83 if has_bom:
84 if m:
85 magic_comment_encoding = m.group(1).decode('latin-1')
86 if magic_comment_encoding != 'utf-8':
87 raise SyntaxError(
88 'encoding problem: {0} with BOM'.format(
89 magic_comment_encoding))
90 return 'utf-8'
91 elif m:
92 return m.group(1).decode('latin-1')
93 else:
94 return None
95 finally:
96 fp.seek(pos)

103 def parse_future_flags(fp, encoding='latin-1'):
104 """Parse the compiler flags by :mod:`__future__` from the given Python
105 code.
106 """
107 import __future__
108 pos = fp.tell()
109 fp.seek(0)
110 flags = 0
111 try:
112 body = fp.read().decode(encoding)
113  
114 # Fix up the source to be (hopefully) parsable by regexpen.
  • E501 Line too long (82 > 79 characters)
115 # This will likely do untoward things if the source code itself is broken.
116  
117 # (1) Fix `import (\n...` to be `import (...`.
118 body = re.sub(r'import\s*\([\r\n]+', 'import (', body)
119 # (2) Join line-ending commas with the next line.
120 body = re.sub(r',\s*[\r\n]+', ', ', body)
121 # (3) Remove backslash line continuations.
122 body = re.sub(r'\\\s*[\r\n]+', ' ', body)
123  
124 for m in PYTHON_FUTURE_IMPORT_re.finditer(body):
125 names = [x.strip().strip('()') for x in m.group(1).split(',')]
126 for name in names:
127 feature = getattr(__future__, name, None)
128 if feature:
129 flags |= feature.compiler_flag
130 finally:
131 fp.seek(pos)
132 return flags

135 def pathmatch(pattern, filename):
136 """Extended pathname pattern matching.
137  
138 This function is similar to what is provided by the ``fnmatch`` module in
139 the Python standard library, but:
140  
141 * can match complete (relative or absolute) path names, and not just file
142 names, and
143 * also supports a convenience pattern ("**") to match files at any
144 directory level.
145  
146 Examples:
147  
148 >>> pathmatch('**.py', 'bar.py')
149 True
150 >>> pathmatch('**.py', 'foo/bar/baz.py')
151 True
152 >>> pathmatch('**.py', 'templates/index.html')
153 False
154  
155 >>> pathmatch('./foo/**.py', 'foo/bar/baz.py')
156 True
157 >>> pathmatch('./foo/**.py', 'bar/baz.py')
158 False
159  
160 >>> pathmatch('^foo/**.py', 'foo/bar/baz.py')
161 True
162 >>> pathmatch('^foo/**.py', 'bar/baz.py')
163 False
164  
165 >>> pathmatch('**/templates/*.html', 'templates/index.html')
166 True
167 >>> pathmatch('**/templates/*.html', 'templates/foo/bar.html')
168 False
169  
170 :param pattern: the glob pattern
171 :param filename: the path name of the file to match against
172 """
173 symbols = {
174 '?': '[^/]',
175 '?/': '[^/]/',
176 '*': '[^/]+',
177 '*/': '[^/]+/',
178 '**/': '(?:.+/)*?',
179 '**': '(?:.+/)*?[^/]+',
180 }
181  
182 if pattern.startswith('^'):
183 buf = ['^']
184 pattern = pattern[1:]
185 elif pattern.startswith('./'):
186 buf = ['^']
187 pattern = pattern[2:]
188 else:
189 buf = []
190  
191 for idx, part in enumerate(re.split('([?*]+/?)', pattern)):
192 if idx % 2:
193 buf.append(symbols[part])
194 elif part:
195 buf.append(re.escape(part))
196 match = re.match(''.join(buf) + '$', filename.replace(os.sep, '/'))
197 return match is not None
