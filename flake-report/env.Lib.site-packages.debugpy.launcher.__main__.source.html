61 # debugpy can also be invoked directly rather than via -m. In this case, the first
62 # entry on sys.path is the one added automatically by Python for the directory
63 # containing this file. This means that import debugpy will not work, since we need
64 # the parent directory of debugpy/ to be in sys.path, rather than debugpy/launcher/.
65 #
66 # The other issue is that many other absolute imports will break, because they
67 # will be resolved relative to debugpy/launcher/ - e.g. `import state` will then try
68 # to import debugpy/launcher/state.py.
69 #
70 # To fix both, we need to replace the automatically added entry such that it points
71 # at parent directory of debugpy/ instead of debugpy/launcher, import debugpy with that
72 # in sys.path, and then remove the first entry entry altogether, so that it doesn't
73 # affect any further imports we might do. For example, suppose the user did:
74 #
75 # python /foo/bar/debugpy/launcher ...
76 #
77 # At the beginning of this script, sys.path will contain "/foo/bar/debugpy/launcher"
78 # as the first entry. What we want is to replace it with "/foo/bar', then import
79 # debugpy with that in effect, and then remove the replaced entry before any more
80 # code runs. The imported debugpy module will remain in sys.modules, and thus all
81 # future imports of it or its submodules will resolve accordingly.
82 if "debugpy" not in sys.modules:
83 # Do not use dirname() to walk up - this can be a relative path, e.g. ".".
84 sys.path[0] = sys.path[0] + "/../../"
85 __import__("debugpy")
86 del sys.path[0]
