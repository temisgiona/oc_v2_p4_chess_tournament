88 def get_sync_start_position(
89 self, document: Document, lineno: int
90 ) -> Tuple[int, int]:
91 """
92 Scan backwards, and find a possible position to start.
93 """
94 pattern = self._compiled_pattern
95 lines = document.lines
96  
97 # Scan upwards, until we find a point where we can start the syntax
98 # synchronisation.
99 for i in range(lineno, max(-1, lineno - self.MAX_BACKWARDS), -1):
100 match = pattern.match(lines[i])
101 if match:
102 return i, match.start()
103  
104 # No synchronisation point found. If we aren't that far from the
105 # beginning, start at the very beginning, otherwise, just try to start
106 # at the current line.
107 if lineno < self.FROM_START_IF_NO_SYNC_POS_FOUND:
108 return 0, 0
109 else:
110 return lineno, 0
