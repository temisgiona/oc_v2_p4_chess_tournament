14 def kleene_or(
15 left: bool | np.ndarray,
16 right: bool | np.ndarray,
17 left_mask: np.ndarray | None,
18 right_mask: np.ndarray | None,
19 ):
20 """
21 Boolean ``or`` using Kleene logic.
22  
23 Values are NA where we have ``NA | NA`` or ``NA | False``.
24 ``NA | True`` is considered True.
25  
26 Parameters
27 ----------
28 left, right : ndarray, NA, or bool
29 The values of the array.
30 left_mask, right_mask : ndarray, optional
31 The masks. Only one of these may be None, which implies that
32 the associated `left` or `right` value is a scalar.
33  
34 Returns
35 -------
36 result, mask: ndarray[bool]
37 The result of the logical or, and the new mask.
38 """
39 # To reduce the number of cases, we ensure that `left` & `left_mask`
40 # always come from an array, not a scalar. This is safe, since because
41 # A | B == B | A
42 if left_mask is None:
43 return kleene_or(right, left, right_mask, left_mask)
44  
45 assert isinstance(left, np.ndarray)
46  
47 raise_for_nan(right, method="or")
48  
49 if right is libmissing.NA:
50 result = left.copy()
51 else:
52 result = left | right
53  
54 if right_mask is not None:
55 # output is unknown where (False & NA), (NA & False), (NA & NA)
56 left_false = ~(left | left_mask)
57 right_false = ~(right | right_mask)
58 mask = (
59 (left_false & right_mask)
60 | (right_false & left_mask)
61 | (left_mask & right_mask)
62 )
63 else:
64 if right is True:
65 mask = np.zeros_like(left_mask)
66 elif right is libmissing.NA:
67 mask = (~left & ~left_mask) | left_mask
68 else:
69 # False
70 mask = left_mask.copy()
71  
72 return result, mask
