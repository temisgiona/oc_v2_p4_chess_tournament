100 def parse(
101 cls,
102 read_exact: Callable[[int], Generator[None, None, bytes]],
103 *,
104 mask: bool,
105 max_size: Optional[int] = None,
106 extensions: Optional[Sequence["extensions.Extension"]] = None,
107 ) -> Generator[None, None, "Frame"]:
108 """
109 Read a WebSocket frame.
110  
111 :param read_exact: generator-based coroutine that reads the requested
112 number of bytes or raises an exception if there isn't enough data
113 :param mask: whether the frame should be masked i.e. whether the read
114 happens on the server side
115 :param max_size: maximum payload size in bytes
116 :param extensions: list of classes with a ``decode()`` method that
117 transforms the frame and return a new frame; extensions are applied
118 in reverse order
119 :raises ~websockets.exceptions.PayloadTooBig: if the frame exceeds
120 ``max_size``
121 :raises ~websockets.exceptions.ProtocolError: if the frame
122 contains incorrect values
123  
124 """
125 # Read the header.
126 data = yield from read_exact(2)
127 head1, head2 = struct.unpack("!BB", data)
128  
129 # While not Pythonic, this is marginally faster than calling bool().
130 fin = True if head1 & 0b10000000 else False
131 rsv1 = True if head1 & 0b01000000 else False
132 rsv2 = True if head1 & 0b00100000 else False
133 rsv3 = True if head1 & 0b00010000 else False
134  
135 try:
136 opcode = Opcode(head1 & 0b00001111)
137 except ValueError as exc:
138 raise ProtocolError("invalid opcode") from exc
139  
140 if (True if head2 & 0b10000000 else False) != mask:
141 raise ProtocolError("incorrect masking")
142  
143 length = head2 & 0b01111111
144 if length == 126:
145 data = yield from read_exact(2)
146 (length,) = struct.unpack("!H", data)
147 elif length == 127:
148 data = yield from read_exact(8)
149 (length,) = struct.unpack("!Q", data)
150 if max_size is not None and length > max_size:
  • E501 Line too long (81 > 79 characters)
151 raise PayloadTooBig(f"over size limit ({length} > {max_size} bytes)")
152 if mask:
153 mask_bytes = yield from read_exact(4)
154  
155 # Read the data.
156 data = yield from read_exact(length)
157 if mask:
158 data = apply_mask(data, mask_bytes)
159  
160 frame = cls(fin, opcode, data, rsv1, rsv2, rsv3)
161  
162 if extensions is None:
163 extensions = []
164 for extension in reversed(extensions):
165 frame = extension.decode(frame, max_size=max_size)
166  
167 frame.check()
168  
169 return frame

171 def serialize(
172 self,
173 *,
174 mask: bool,
175 extensions: Optional[Sequence["extensions.Extension"]] = None,
176 ) -> bytes:
177 """
178 Write a WebSocket frame.
179  
180 :param frame: frame to write
181 :param mask: whether the frame should be masked i.e. whether the write
182 happens on the client side
183 :param extensions: list of classes with an ``encode()`` method that
184 transform the frame and return a new frame; extensions are applied
185 in order
186 :raises ~websockets.exceptions.ProtocolError: if the frame
187 contains incorrect values
188  
189 """
190 self.check()
191  
192 if extensions is None:
193 extensions = []
194 for extension in extensions:
195 self = extension.encode(self)
196  
197 output = io.BytesIO()
198  
199 # Prepare the header.
200 head1 = (
201 (0b10000000 if self.fin else 0)
202 | (0b01000000 if self.rsv1 else 0)
203 | (0b00100000 if self.rsv2 else 0)
204 | (0b00010000 if self.rsv3 else 0)
205 | self.opcode
206 )
207  
208 head2 = 0b10000000 if mask else 0
209  
210 length = len(self.data)
211 if length < 126:
212 output.write(struct.pack("!BB", head1, head2 | length))
213 elif length < 65536:
214 output.write(struct.pack("!BBH", head1, head2 | 126, length))
215 else:
216 output.write(struct.pack("!BBQ", head1, head2 | 127, length))
217  
218 if mask:
219 mask_bytes = secrets.token_bytes(4)
220 output.write(mask_bytes)
221  
222 # Prepare the data.
223 if mask:
224 data = apply_mask(self.data, mask_bytes)
225 else:
226 data = self.data
227 output.write(data)
228  
229 return output.getvalue()

231 def check(self) -> None:
232 """
233 Check that reserved bits and opcode have acceptable values.
234  
235 :raises ~websockets.exceptions.ProtocolError: if a reserved
236 bit or the opcode is invalid
237  
238 """
239 if self.rsv1 or self.rsv2 or self.rsv3:
240 raise ProtocolError("reserved bits must be 0")
241  
242 if self.opcode in CTRL_OPCODES:
243 if len(self.data) > 125:
244 raise ProtocolError("control frame too long")
245 if not self.fin:
246 raise ProtocolError("fragmented control frame")

249 def prepare_data(data: Data) -> Tuple[int, bytes]:
250 """
251 Convert a string or byte-like object to an opcode and a bytes-like object.
252  
253 This function is designed for data frames.
254  
255 If ``data`` is a :class:`str`, return ``OP_TEXT`` and a :class:`bytes`
256 object encoding ``data`` in UTF-8.
257  
258 If ``data`` is a bytes-like object, return ``OP_BINARY`` and a bytes-like
259 object.
260  
261 :raises TypeError: if ``data`` doesn't have a supported type
262  
263 """
264 if isinstance(data, str):
265 return OP_TEXT, data.encode("utf-8")
266 elif isinstance(data, (bytes, bytearray, memoryview)):
267 return OP_BINARY, data
268 else:
269 raise TypeError("data must be bytes-like or str")

294 def parse_close(data: bytes) -> Tuple[int, str]:
295 """
296 Parse the payload from a close frame.
297  
298 Return ``(code, reason)``.
299  
300 :raises ~websockets.exceptions.ProtocolError: if data is ill-formed
301 :raises UnicodeDecodeError: if the reason isn't valid UTF-8
302  
303 """
304 length = len(data)
305 if length >= 2:
306 (code,) = struct.unpack("!H", data[:2])
307 check_close(code)
308 reason = data[2:].decode("utf-8")
309 return code, reason
310 elif length == 0:
311 return 1005, ""
312 else:
313 assert length == 1
314 raise ProtocolError("close frame too short")
