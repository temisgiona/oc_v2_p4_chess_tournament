101 def _get_match(self, prefix: str) -> Union[None, Keys, Tuple[Keys, ...]]:
102 """
103 Return the key (or keys) that maps to this prefix.
104 """
105 # (hard coded) If we match a CPR response, return Keys.CPRResponse.
106 # (This one doesn't fit in the ANSI_SEQUENCES, because it contains
107 # integer variables.)
108 if _cpr_response_re.match(prefix):
109 return Keys.CPRResponse
110  
111 elif _mouse_event_re.match(prefix):
112 return Keys.Vt100MouseEvent
113  
114 # Otherwise, use the mappings.
115 try:
116 return ANSI_SEQUENCES[prefix]
117 except KeyError:
118 return None

228 def flush(self) -> None:
229 """
230 Flush the buffer of the input stream.
231  
232 This will allow us to handle the escape key (or maybe meta) sooner.
233 The input received by the escape key is actually the same as the first
234 characters of e.g. Arrow-Up, so without knowing what follows the escape
235 sequence, we don't know whether escape has been pressed, or whether
236 it's something else. This flush function should be called after a
237 timeout, and processes everything that's still in the buffer as-is, so
238 without assuming any characters will follow.
239 """
240 self._input_parser.send(_Flush())
