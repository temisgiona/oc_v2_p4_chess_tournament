114 # In Python 2, urllib.unquote doesn't handle Unicode strings correctly,
115 # so we need to convert to ASCII first, unquote, and then decode.
116 def unquote(s):
117 if s is None:
118 return None
119 if not isinstance(s, bytes):
120 s = bytes(s)
121 s = urllib.unquote(s)
122 if isinstance(s, bytes):
123 s = s.decode('utf-8')
124 return s

166 def _parse_debug_options(opts):
167 """Debug options are semicolon separated key=value pairs
168 """
169 options = {}
170 if not opts:
171 return options
172  
173 for opt in opts.split(';'):
174 try:
175 key, value = opt.split('=')
176 except ValueError:
177 continue
178 try:
179 options[key] = DEBUG_OPTIONS_PARSER[key](value)
180 except KeyError:
181 continue
182  
183 return options

186 def _extract_debug_options(opts, flags=None):
187 """Return the debug options encoded in the given value.
188  
189 "opts" is a semicolon-separated string of "key=value" pairs.
190 "flags" is a list of strings.
191  
192 If flags is provided then it is used as a fallback.
193  
194 The values come from the launch config:
195

197 type:'python',
198 request:'launch'|'attach',
199 name:'friendly name for debug config',

206 Further information can be found here:
207

210 if not opts:
