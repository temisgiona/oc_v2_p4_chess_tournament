46 class _Socket(_original_Socket):
47 """Green version of :class:`zmq.Socket`
48  
49 The following methods are overridden:
50  
51 * send
52 * recv
53  
54 To ensure that the ``zmq.NOBLOCK`` flag is set and that sending or receiving
55 is deferred to the hub if a ``zmq.EAGAIN`` (retry) error is raised.
56  
57 The `__state_changed` method is triggered when the zmq.FD for the socket is
58 marked as readable and triggers the necessary read and write events (which
59 are waited for in the recv and send methods).
60  
61 Some double underscore prefixes are used to minimize pollution of
62 :class:`zmq.Socket`'s namespace.
63 """
64  
65 __in_send_multipart = False
66 __in_recv_multipart = False
67 __writable = None
68 __readable = None
69 _state_event = None
70 _gevent_bug_timeout = 11.6 # timeout for not trusting gevent
71 _debug_gevent = False # turn on if you think gevent is missing events
72 _poller_class = _Poller

87 def __cleanup_events(self):
88 # close the _state_event event, keeps the number of active file descriptors down
89 if getattr(self, '_state_event', None):
90 _stop(self._state_event)
91 self._state_event = None
92 # if the socket has entered a close state resume any waiting greenlets
93 self.__writable.set()
94 self.__readable.set()

96 def __setup_events(self):
97 self.__readable = AsyncResult()
98 self.__writable = AsyncResult()
99 self.__readable.set()
100 self.__writable.set()

102 try:
103 self._state_event = get_hub().loop.io(
104 self.getsockopt(zmq.FD), 1
105 ) # read state watcher
106 self._state_event.start(self.__state_changed)
107 except AttributeError:
108 # for gevent<1.0 compatibility
109 from gevent.core import read_event
110  
111 self._state_event = read_event(
112 self.getsockopt(zmq.FD), self.__state_changed, persist=True

115 def __state_changed(self, event=None, _evtype=None):
116 if self.closed:
117 self.__cleanup_events()
118 return
119 try:
120 # avoid triggering __state_changed from inside __state_changed
121 events = super(_Socket, self).getsockopt(zmq.EVENTS)
122 except zmq.ZMQError as exc:
123 self.__writable.set_exception(exc)
124 self.__readable.set_exception(exc)
125 else:
126 if events & zmq.POLLOUT:
127 self.__writable.set()
128 if events & zmq.POLLIN:
129 self.__readable.set()

131 def _wait_write(self):
132 assert self.__writable.ready(), "Only one greenlet can be waiting on this event"
133 self.__writable = AsyncResult()
134 # timeout is because libzmq cannot be trusted to properly signal a new send event:
135 # this is effectively a maximum poll interval of 1s
136 tic = time.time()
137 dt = self._gevent_bug_timeout
138 if dt:
139 timeout = gevent.Timeout(seconds=dt)
140 else:
141 timeout = None
142 try:
143 if timeout:
144 timeout.start()
145 self.__writable.get(block=True)
146 except gevent.Timeout as t:
147 if t is not timeout:
148 raise
149 toc = time.time()
150 # gevent bug: get can raise timeout even on clean return
151 # don't display zmq bug warning for gevent bug (this is getting ridiculous)
152 if (
153 self._debug_gevent
154 and timeout
155 and toc - tic > dt
156 and self.getsockopt(zmq.EVENTS) & zmq.POLLOUT
157 ):
158 print(
159 "BUG: gevent may have missed a libzmq send event on %i!" % self.FD,
160 file=sys.stderr,
161 )
162 finally:
163 if timeout:
164 timeout.cancel()
165 self.__writable.set()

167 def _wait_read(self):
168 assert self.__readable.ready(), "Only one greenlet can be waiting on this event"
169 self.__readable = AsyncResult()
170 # timeout is because libzmq cannot always be trusted to play nice with libevent.
171 # I can only confirm that this actually happens for send, but lets be symmetrical
172 # with our dirty hacks.
173 # this is effectively a maximum poll interval of 1s
174 tic = time.time()
175 dt = self._gevent_bug_timeout
176 if dt:
177 timeout = gevent.Timeout(seconds=dt)
178 else:
179 timeout = None
180 try:
181 if timeout:
182 timeout.start()
183 self.__readable.get(block=True)
184 except gevent.Timeout as t:
185 if t is not timeout:
186 raise
187 toc = time.time()
188 # gevent bug: get can raise timeout even on clean return

190 if (
191 self._debug_gevent
192 and timeout
193 and toc - tic > dt
194 and self.getsockopt(zmq.EVENTS) & zmq.POLLIN
195 ):
196 print(
197 "BUG: gevent may have missed a libzmq recv event on %i!" % self.FD,

205 def send(self, data, flags=0, copy=True, track=False, **kwargs):
206 """send, which will only block current greenlet
207  
208 state_changed always fires exactly once (success or fail) at the
209 end of this method.
210 """
211  
212 # if we're given the NOBLOCK flag act as normal and let the EAGAIN get raised
213 if flags & zmq.NOBLOCK:
214 try:
215 msg = super(_Socket, self).send(data, flags, copy, track, **kwargs)
216 finally:
217 if not self.__in_send_multipart:
218 self.__state_changed()
219 return msg
220 # ensure the zmq.NOBLOCK flag is part of flags
221 flags |= zmq.NOBLOCK
222 while (
223 True
224 ): # Attempt to complete this operation indefinitely, blocking the current greenlet
225 try:
226 # attempt the actual call
227 msg = super(_Socket, self).send(data, flags, copy, track)
228 except zmq.ZMQError as e:
229 # if the raised ZMQError is not EAGAIN, reraise
230 if e.errno != zmq.EAGAIN:
231 if not self.__in_send_multipart:
232 self.__state_changed()
233 raise
234 else:
235 if not self.__in_send_multipart:
236 self.__state_changed()
237 return msg
238 # defer to the event loop until we're notified the socket is writable
239 self._wait_write()

241 def recv(self, flags=0, copy=True, track=False):
242 """recv, which will only block current greenlet
243  
244 state_changed always fires exactly once (success or fail) at the
245 end of this method.
246 """
247 if flags & zmq.NOBLOCK:
248 try:
249 msg = super(_Socket, self).recv(flags, copy, track)
250 finally:
251 if not self.__in_recv_multipart:
252 self.__state_changed()
253 return msg
254  
255 flags |= zmq.NOBLOCK
256 while True:
257 try:
258 msg = super(_Socket, self).recv(flags, copy, track)
259 except zmq.ZMQError as e:
260 if e.errno != zmq.EAGAIN:
261 if not self.__in_recv_multipart:
262 self.__state_changed()
263 raise
264 else:
265 if not self.__in_recv_multipart:
266 self.__state_changed()
267 return msg
268 self._wait_read()

270 def send_multipart(self, *args, **kwargs):
271 """wrap send_multipart to prevent state_changed on each partial send"""
272 self.__in_send_multipart = True
273 try:
274 msg = super(_Socket, self).send_multipart(*args, **kwargs)
275 finally:
276 self.__in_send_multipart = False
277 self.__state_changed()
278 return msg

290 def get(self, opt):
291 """trigger state_changed on getsockopt(EVENTS)"""
292 if opt in TIMEOS:
293 warnings.warn(
294 "TIMEO socket options have no effect in zmq.green", UserWarning
295 )
296 optval = super(_Socket, self).get(opt)
297 if opt == zmq.EVENTS:
298 self.__state_changed()
299 return optval
