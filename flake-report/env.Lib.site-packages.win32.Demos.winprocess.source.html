44 def __init__(self, cmd, login=None,
45 hStdin=None, hStdout=None, hStderr=None,
46 show=1, xy=None, xySize=None,
47 desktop=None):
48 """
49 Create a Windows process.
50 cmd: command to run
51 login: run as user 'Domain\nUser\nPassword'
52 hStdin, hStdout, hStderr:
53 handles for process I/O; default is caller's stdin,
54 stdout & stderr
55 show: wShowWindow (0=SW_HIDE, 1=SW_NORMAL, ...)
56 xy: window offset (x, y) of upper left corner in pixels
57 xySize: window size (width, height) in pixels
58 desktop: lpDesktop - name of desktop e.g. 'winsta0\\default'
59 None = inherit current desktop
60 '' = create new desktop if necessary
61  
62 User calling login requires additional privileges:
63 Act as part of the operating system [not needed on Windows XP]
64 Increase quotas
65 Replace a process level token
66 Login string must EITHER be an administrator's account
67 (ordinary user can't access current desktop - see Microsoft
68 Q165194) OR use desktop='' to run another desktop invisibly
69 (may be very slow to startup & finalize).
70 """
71 si = win32process.STARTUPINFO()
72 si.dwFlags = (win32con.STARTF_USESTDHANDLES ^
73 win32con.STARTF_USESHOWWINDOW)
74 if hStdin is None:
75 si.hStdInput = win32api.GetStdHandle(win32api.STD_INPUT_HANDLE)
76 else:
77 si.hStdInput = hStdin
78 if hStdout is None:
79 si.hStdOutput = win32api.GetStdHandle(win32api.STD_OUTPUT_HANDLE)
80 else:
81 si.hStdOutput = hStdout
82 if hStderr is None:
83 si.hStdError = win32api.GetStdHandle(win32api.STD_ERROR_HANDLE)
84 else:
85 si.hStdError = hStderr
86 si.wShowWindow = show
87 if xy is not None:
88 si.dwX, si.dwY = xy
89 si.dwFlags ^= win32con.STARTF_USEPOSITION
90 if xySize is not None:
91 si.dwXSize, si.dwYSize = xySize
92 si.dwFlags ^= win32con.STARTF_USESIZE
93 if desktop is not None:
94 si.lpDesktop = desktop
95 procArgs = (None, # appName
96 cmd, # commandLine
97 None, # processAttributes
98 None, # threadAttributes
99 1, # bInheritHandles
100 win32process.CREATE_NEW_CONSOLE, # dwCreationFlags
101 None, # newEnvironment
102 None, # currentDirectory
103 si) # startupinfo
104 if login is not None:
105 hUser = logonUser(login)
106 win32security.ImpersonateLoggedOnUser(hUser)
107 procHandles = win32process.CreateProcessAsUser(hUser, *procArgs)
108 win32security.RevertToSelf()
109 else:
110 procHandles = win32process.CreateProcess(*procArgs)
111 self.hProcess, self.hThread, self.PId, self.TId = procHandles

123 """
124 Kill process. Try for an orderly shutdown via WM_CLOSE. If

148 def run(cmd, mSec=None, stdin=None, stdout=None, stderr=None, **kw):
149 """
150 Run cmd as a child process and return exit code.
151 mSec: terminate cmd after specified number of milliseconds
152 stdin, stdout, stderr:
153 file objects for child I/O (use hStdin etc. to attach
154 handles instead of files); default is caller's stdin,
155 stdout & stderr;
156 kw: see Process.__init__ for more keyword options
157 """
158 if stdin is not None:
159 kw['hStdin'] = msvcrt.get_osfhandle(stdin.fileno())
160 if stdout is not None:
161 kw['hStdout'] = msvcrt.get_osfhandle(stdout.fileno())
162 if stderr is not None:
163 kw['hStderr'] = msvcrt.get_osfhandle(stderr.fileno())
164 child = Process(cmd, **kw)
165 if child.wait(mSec) != win32event.WAIT_OBJECT_0:
166 child.kill()
167 raise WindowsError('process timeout exceeded')
168 return child.exitCode()
