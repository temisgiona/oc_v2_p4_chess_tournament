171 def _handle_truncated_float_vec(vec, nbytes):
172 # This feature is not well documented, but some SAS XPORT files
173 # have 2-7 byte "truncated" floats. To read these truncated
174 # floats, pad them with zeros on the right to make 8 byte floats.
175 #
176 # References:
177 # https://github.com/jcushman/xport/pull/3
178 # The R "foreign" library
179  
180 if nbytes != 8:
181 vec1 = np.zeros(len(vec), np.dtype("S8"))
182 dtype = np.dtype(f"S{nbytes},S{8 - nbytes}")
183 vec2 = vec1.view(dtype=dtype)
184 vec2["f0"] = vec
185 return vec2
186  
187 return vec

190 def _parse_float_vec(vec):
191 """
192 Parse a vector of float values representing IBM 8 byte floats into
193 native 8 byte floats.
194 """
195 dtype = np.dtype(">u4,>u4")
196 vec1 = vec.view(dtype=dtype)
197 xport1 = vec1["f0"]
198 xport2 = vec1["f1"]
199  
200 # Start by setting first half of ieee number to first half of IBM
201 # number sans exponent
202 ieee1 = xport1 & 0x00FFFFFF
203  
204 # The fraction bit to the left of the binary point in the ieee
205 # format was set and the number was shifted 0, 1, 2, or 3
206 # places. This will tell us how to adjust the ibm exponent to be a
207 # power of 2 ieee exponent and how to shift the fraction bits to
208 # restore the correct magnitude.
209 shift = np.zeros(len(vec), dtype=np.uint8)
210 shift[np.where(xport1 & 0x00200000)] = 1
211 shift[np.where(xport1 & 0x00400000)] = 2
212 shift[np.where(xport1 & 0x00800000)] = 3
213  
214 # shift the ieee number down the correct number of places then
215 # set the second half of the ieee number to be the second half
216 # of the ibm number shifted appropriately, ored with the bits
217 # from the first half that would have been shifted in if we
218 # could shift a double. All we are worried about are the low
219 # order 3 bits of the first half since we're only shifting by
220 # 1, 2, or 3.
221 ieee1 >>= shift
222 ieee2 = (xport2 >> shift) | ((xport1 & 0x00000007) << (29 + (3 - shift)))
223  
224 # clear the 1 bit to the left of the binary point
225 ieee1 &= 0xFFEFFFFF
226  
227 # set the exponent of the ieee number to be the actual exponent
228 # plus the shift count + 1023. Or this into the first half of the
229 # ieee number. The ibm exponent is excess 64 but is adjusted by 65
230 # since during conversion to ibm format the exponent is
231 # incremented by 1 and the fraction bits left 4 positions to the
232 # right of the radix point. (had to add >> 24 because C treats &
233 # 0x7f as 0x7f000000 and Python doesn't)
234 ieee1 |= ((((((xport1 >> 24) & 0x7F) - 65) << 2) + shift + 1023) << 20) | (
235 xport1 & 0x80000000
236 )
237  
238 ieee = np.empty((len(ieee1),), dtype=">u4,>u4")
239 ieee["f0"] = ieee1
240 ieee["f1"] = ieee2
241 ieee = ieee.view(dtype=">f8")
242 ieee = ieee.astype("f8")
243  
244 return ieee

276 def _read_header(self):
277 self.filepath_or_buffer.seek(0)
278  
279 # read file header
280 line1 = self._get_row()
281 if line1 != _correct_line1:
282 raise ValueError("Header record is not an XPORT file.")
283  
284 line2 = self._get_row()
285 fif = [["prefix", 24], ["version", 8], ["OS", 8], ["_", 24], ["created", 16]]
286 file_info = _split_line(line2, fif)
287 if file_info["prefix"] != "SAS SAS SASLIB":
288 raise ValueError("Header record has invalid prefix.")
289 file_info["created"] = _parse_date(file_info["created"])
290 self.file_info = file_info
291  
292 line3 = self._get_row()
293 file_info["modified"] = _parse_date(line3[:16])
294  
295 # read member header
296 header1 = self._get_row()
297 header2 = self._get_row()
298 headflag1 = header1.startswith(_correct_header1)
299 headflag2 = header2 == _correct_header2
300 if not (headflag1 and headflag2):
301 raise ValueError("Member header not found")
302 # usually 140, could be 135
303 fieldnamelength = int(header1[-5:-2])
304  
305 # member info
306 mem = [
307 ["prefix", 8],
308 ["set_name", 8],
309 ["sasdata", 8],
310 ["version", 8],
311 ["OS", 8],
312 ["_", 24],
313 ["created", 16],
314 ]
315 member_info = _split_line(self._get_row(), mem)
316 mem = [["modified", 16], ["_", 16], ["label", 40], ["type", 8]]
317 member_info.update(_split_line(self._get_row(), mem))
318 member_info["modified"] = _parse_date(member_info["modified"])
319 member_info["created"] = _parse_date(member_info["created"])
320 self.member_info = member_info
321  
322 # read field names
323 types = {1: "numeric", 2: "char"}
324 fieldcount = int(self._get_row()[54:58])
325 datalength = fieldnamelength * fieldcount
326 # round up to nearest 80
327 if datalength % 80:
328 datalength += 80 - datalength % 80
329 fielddata = self.filepath_or_buffer.read(datalength)
330 fields = []
331 obs_length = 0
332 while len(fielddata) >= fieldnamelength:
333 # pull data for one field
334 fieldbytes, fielddata = (
335 fielddata[:fieldnamelength],
336 fielddata[fieldnamelength:],
337 )
338  
339 # rest at end gets ignored, so if field is short, pad out
340 # to match struct pattern below
341 fieldbytes = fieldbytes.ljust(140)
342  
343 fieldstruct = struct.unpack(">hhhh8s40s8shhh2s8shhl52s", fieldbytes)
344 field = dict(zip(_fieldkeys, fieldstruct))
345 del field["_"]
346 field["ntype"] = types[field["ntype"]]
347 fl = field["field_length"]
348 if field["ntype"] == "numeric" and ((fl < 2) or (fl > 8)):
349 msg = f"Floating field width {fl} is not between 2 and 8."
350 raise TypeError(msg)
351  
352 for k, v in field.items():
353 try:
354 field[k] = v.strip()
355 except AttributeError:
356 pass
357  
358 obs_length += field["field_length"]
359 fields += [field]
360  
361 header = self._get_row()
362 if not header == _correct_obs_header:
363 raise ValueError("Observation header not found.")
364  
365 self.fields = fields
366 self.record_length = obs_length
367 self.record_start = self.filepath_or_buffer.tell()
368  
369 self.nobs = self._record_count()
370 self.columns = [x["name"].decode() for x in self.fields]
371  
372 # Setup the dtype.
373 dtypel = [
374 ("s" + str(i), "S" + str(field["field_length"]))
375 for i, field in enumerate(self.fields)
376 ]
377 dtype = np.dtype(dtypel)
378 self._dtype = dtype

383 def _record_count(self) -> int:
384 """
385 Get number of records in file.
386  
387 This is maybe suboptimal because we have to seek to the end of
388 the file.
389  
390 Side effect: returns file position to record_start.
391 """
392 self.filepath_or_buffer.seek(0, 2)
393 total_records_length = self.filepath_or_buffer.tell() - self.record_start
394  
395 if total_records_length % 80 != 0:
396 warnings.warn("xport file may be corrupted")
397  
398 if self.record_length > 80:
399 self.filepath_or_buffer.seek(self.record_start)
400 return total_records_length // self.record_length
401  
402 self.filepath_or_buffer.seek(-80, 2)
403 last_card_bytes = self.filepath_or_buffer.read(80)
404 last_card = np.frombuffer(last_card_bytes, dtype=np.uint64)
405  
406 # 8 byte blank
407 ix = np.flatnonzero(last_card == 2314885530818453536)
408  
409 if len(ix) == 0:
410 tail_pad = 0
411 else:
412 tail_pad = 8 * len(ix)
413  
414 self.filepath_or_buffer.seek(self.record_start)
415  
416 return (total_records_length - tail_pad) // self.record_length

435 def _missing_double(self, vec):
436 v = vec.view(dtype="u1,u1,u2,u4")
437 miss = (v["f1"] == 0) & (v["f2"] == 0) & (v["f3"] == 0)
438 miss1 = (
439 ((v["f0"] >= 0x41) & (v["f0"] <= 0x5A))
440 | (v["f0"] == 0x5F)
441 | (v["f0"] == 0x2E)
442 )
443 miss &= miss1
444 return miss

446 @Appender(_read_method_doc)
447 def read(self, nrows=None):
448  
449 if nrows is None:
450 nrows = self.nobs
451  
452 read_lines = min(nrows, self.nobs - self._lines_read)
453 read_len = read_lines * self.record_length
454 if read_len <= 0:
455 self.close()
456 raise StopIteration
457 raw = self.filepath_or_buffer.read(read_len)
458 data = np.frombuffer(raw, dtype=self._dtype, count=read_lines)
459  
460 df = pd.DataFrame(index=range(read_lines))
461 for j, x in enumerate(self.columns):
462 vec = data["s" + str(j)]
463 ntype = self.fields[j]["ntype"]
464 if ntype == "numeric":
465 vec = _handle_truncated_float_vec(vec, self.fields[j]["field_length"])
466 miss = self._missing_double(vec)
467 v = _parse_float_vec(vec)
468 v[miss] = np.nan
469 elif self.fields[j]["ntype"] == "char":
470 v = [y.rstrip() for y in vec]
471  
472 if self._encoding is not None:
473 v = [y.decode(self._encoding) for y in v]
474  
475 df[x] = v
476  
477 if self._index is None:
478 df.index = pd.Index(range(self._lines_read, self._lines_read + read_lines))
479 else:
480 df = df.set_index(self._index)
481  
482 self._lines_read += read_lines
483  
484 return df
