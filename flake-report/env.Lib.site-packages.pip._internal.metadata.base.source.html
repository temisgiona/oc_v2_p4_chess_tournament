280 local_only: bool = True,
281 skip: Container[str] = stdlib_pkgs,
282 include_editables: bool = True,
283 editables_only: bool = False,
284 user_only: bool = False,
285 ) -> Iterator[BaseDistribution]:
286 """Return a list of installed distributions.
287  
288 :param local_only: If True (default), only return installations
289 local to the current virtualenv, if in a virtualenv.
290 :param skip: An iterable of canonicalized project names to ignore;
291 defaults to ``stdlib_pkgs``.
292 :param include_editables: If False, don't report editables.
293 :param editables_only: If True, only report editables.
294 :param user_only: If True, only report installations in the user
295 site directory.
296 """
297 it = self.iter_distributions()
298 if local_only:
299 it = (d for d in it if d.local)
300 if not include_editables:
301 it = (d for d in it if not d.editable)
302 if editables_only:
303 it = (d for d in it if d.editable)
304 if user_only:
305 it = (d for d in it if d.in_usersite)
306 return (d for d in it if d.canonical_name not in skip)
