165 def _calculate_sha256_for_file(path):
166 sha256 = hashlib.sha256()
167 with open(path, 'rb') as f:
168 for block in iter(lambda: f.read(filecmp.BUFSIZE), b''):
169 sha256.update(block)
170 return sha256.hexdigest()

194 def _try_get_same_env():
195 env = SameEnvironment()
196 if not os.path.basename(env.executable).lower().startswith('python'):
197 # This tries to counter issues with embedding. In some cases (e.g.
198 # VIM's Python Mac/Windows, sys.executable is /foo/bar/vim. This
199 # happens, because for Mac a function called `_NSGetExecutablePath` is
200 # used and for Windows `GetModuleFileNameW`. These are both platform
201 # specific functions. For all other systems sys.executable should be
202 # alright. However here we try to generalize:
203 #
204 # 1. Check if the executable looks like python (heuristic)
205 # 2. In case it's not try to find the executable
206 # 3. In case we don't find it use an interpreter environment.
207 #
208 # The last option will always work, but leads to potential crashes of
209 # Jedi - which is ok, because it happens very rarely and even less,
210 # because the code below should work for most cases.
211 if os.name == 'nt':
212 # The first case would be a virtualenv and the second a normal
213 # Python installation.
214 checks = (r'Scripts\python.exe', 'python.exe')
215 else:
216 # For unix it looks like Python is always in a bin folder.
217 checks = (
218 'bin/python%s.%s' % (sys.version_info[0], sys.version[1]),
219 'bin/python%s' % (sys.version_info[0]),
220 'bin/python',
221 )
222 for check in checks:
223 guess = os.path.join(sys.exec_prefix, check)
224 if os.path.isfile(guess):
225 # Bingo - We think we have our Python.
226 return Environment(guess)
227 # It looks like there is no reasonable Python to be found.
228 return InterpreterEnvironment()
229 # If no virtualenv is found, use the environment we're already
230 # using.
231 return env

416 def _is_safe(executable_path):
417 # Resolve sym links. A venv typically is a symlink to a known Python
418 # binary. Only virtualenvs copy symlinks around.
419 real_path = os.path.realpath(executable_path)
420  
421 if _is_unix_safe_simple(real_path):
422 return True
423  
424 # Just check the list of known Python versions. If it's not in there,
425 # it's likely an attacker or some Python that was not properly
426 # installed in the system.
427 for environment in find_system_environments():
428 if environment.executable == real_path:
429 return True
430  
431 # If the versions don't match, just compare the binary files. If we
432 # don't do that, only venvs will be working and not virtualenvs.
433 # venvs are symlinks while virtualenvs are actual copies of the
434 # Python files.
435 # This still means that if the system Python is updated and the
436 # virtualenv's Python is not (which is probably never going to get
437 # upgraded), it will not work with Jedi. IMO that's fine, because
438 # people should just be using venv. ~ dave
439 if environment._sha256 == _calculate_sha256_for_file(real_path):
440 return True
441 return False

444 def _is_unix_safe_simple(real_path):
445 if _is_unix_admin():
446 # In case we are root, just be conservative and
447 # only execute known paths.
448 return any(real_path.startswith(p) for p in _SAFE_PATHS)
449  
450 uid = os.stat(real_path).st_uid
451 # The interpreter needs to be owned by root. This means that it wasn't
452 # written by a user and therefore attacking Jedi is not as simple.
453 # The attack could look like the following:
454 # 1. A user clones a repository.
455 # 2. The repository has an innocent looking folder called foobar. jedi
456 # searches for the folder and executes foobar/bin/python --version if
457 # there's also a foobar/bin/activate.
458 # 3. The attacker has gained code execution, since he controls
459 # foobar/bin/python.
460 return uid == 0
