164 def _get_ax_freq(ax: Axes):
165 """
166 Get the freq attribute of the ax object if set.
167 Also checks shared axes (eg when using secondary yaxis, sharex=True
168 or twinx)
169 """
170 ax_freq = getattr(ax, "freq", None)
171 if ax_freq is None:
172 # check for left/right ax in case of secondary yaxis
173 if hasattr(ax, "left_ax"):
174 ax_freq = getattr(ax.left_ax, "freq", None)
175 elif hasattr(ax, "right_ax"):
176 ax_freq = getattr(ax.right_ax, "freq", None)
177 if ax_freq is None:
178 # check if a shared ax (sharex/twinx) has already freq set
179 shared_axes = ax.get_shared_x_axes().get_siblings(ax)
180 if len(shared_axes) > 1:
181 for shared_ax in shared_axes:
182 ax_freq = getattr(shared_ax, "freq", None)
183 if ax_freq is not None:
184 break
185 return ax_freq
