350 def test_to_records_dict_like(self):
351 # see GH#18146
352 class DictLike:
353 def __init__(self, **kwargs):
354 self.d = kwargs.copy()
355  
356 def __getitem__(self, key):
357 return self.d.__getitem__(key)
358  
359 def __contains__(self, key) -> bool:
360 return key in self.d
361  
362 def keys(self):
363 return self.d.keys()
364  
365 df = DataFrame({"A": [1, 2], "B": [0.2, 1.5], "C": ["a", "bc"]})
366  
367 dtype_mappings = {
368 "column_dtypes": DictLike(**{"A": np.int8, "B": np.float32}),
369 "index_dtypes": "<U2",
370 }
371  
372 result = df.to_records(**dtype_mappings)
373 expected = np.rec.array(
374 [("0", "1", "0.2", "a"), ("1", "2", "1.5", "bc")],
375 dtype=[("index", "<U2"), ("A", "i1"), ("B", "<f4"), ("C", "O")],
376 )
377 tm.assert_almost_equal(result, expected)
