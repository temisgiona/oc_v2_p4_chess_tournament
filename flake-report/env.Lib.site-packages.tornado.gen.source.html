1 """``tornado.gen`` implements generator-based coroutines.
2  
3 .. note::
4  
5 The "decorator and generator" approach in this module is a
6 precursor to native coroutines (using ``async def`` and ``await``)
7 which were introduced in Python 3.5. Applications that do not
8 require compatibility with older versions of Python should use
9 native coroutines instead. Some parts of this module are still
10 useful with native coroutines, notably `multi`, `sleep`,
11 `WaitIterator`, and `with_timeout`. Some of these functions have
12 counterparts in the `asyncio` module which may be used as well,
13 although the two may not necessarily be 100% compatible.
14  
15 Coroutines provide an easier way to work in an asynchronous
16 environment than chaining callbacks. Code using coroutines is
17 technically asynchronous, but it is written as a single generator
18 instead of a collection of separate functions.
19  
20 For example, here's a coroutine-based handler:
21  
22 .. testcode::
23  
24 class GenAsyncHandler(RequestHandler):
25 @gen.coroutine
26 def get(self):
27 http_client = AsyncHTTPClient()
28 response = yield http_client.fetch("http://example.com")
29 do_something_with_response(response)
30 self.render("template.html")
31  
32 .. testoutput::
33 :hide:
34  
35 Asynchronous functions in Tornado return an ``Awaitable`` or `.Future`;
36 yielding this object returns its result.
37  
38 You can also yield a list or dict of other yieldable objects, which
39 will be started at the same time and run in parallel; a list or dict
40 of results will be returned when they are all finished:
41  
42 .. testcode::
43  
44 @gen.coroutine
45 def get(self):
46 http_client = AsyncHTTPClient()

48 http_client.fetch(url2)]

50 response4=http_client.fetch(url4))
51 response3 = response_dict['response3']
52 response4 = response_dict['response4']

312

320 expression will raise that exception and all the results will be

719 Maintains information about pending callbacks and their results.
720  
721 The results of the generator are stored in ``result_future`` (a
722 `.Future`)
723 """
724  
725 def __init__(
726 self,
727 ctx_run: Callable,
728 gen: "Generator[_Yieldable, Any, _T]",
729 result_future: "Future[_T]",
730 first_yielded: _Yieldable,
731 ) -> None:
732 self.ctx_run = ctx_run
733 self.gen = gen
734 self.result_future = result_future
735 self.future = _null_future # type: Union[None, Future]
736 self.running = False
737 self.finished = False
738 self.io_loop = IOLoop.current()
739 if self.handle_yield(first_yielded):
740 gen = result_future = first_yielded = None # type: ignore
741 self.ctx_run(self.run)
742  
743 def run(self) -> None:
744 """Starts or resumes the generator, running until it reaches a
745 yield point that is not ready.
746 """
747 if self.running or self.finished:
748 return
749 try:
750 self.running = True
751 while True:
752 future = self.future
753 if future is None:
754 raise Exception("No pending future")
755 if not future.done():
756 return
757 self.future = None
758 try:
759 exc_info = None
760  
761 try:
762 value = future.result()
763 except Exception:
764 exc_info = sys.exc_info()
765 future = None
766  
767 if exc_info is not None:
768 try:
769 yielded = self.gen.throw(*exc_info) # type: ignore
770 finally:
771 # Break up a reference to itself
772 # for faster GC on CPython.
773 exc_info = None
774 else:
775 yielded = self.gen.send(value)
776  
777 except (StopIteration, Return) as e:
778 self.finished = True
779 self.future = _null_future
780 future_set_result_unless_cancelled(
781 self.result_future, _value_from_stopiteration(e)
782 )
783 self.result_future = None # type: ignore
784 return
785 except Exception:
786 self.finished = True
787 self.future = _null_future
788 future_set_exc_info(self.result_future, sys.exc_info())
789 self.result_future = None # type: ignore
790 return
791 if not self.handle_yield(yielded):
792 return
793 yielded = None
794 finally:
795 self.running = False
796  
797 def handle_yield(self, yielded: _Yieldable) -> bool:
798 try:
799 self.future = convert_yielded(yielded)
800 except BadYieldError:
801 self.future = Future()
802 future_set_exc_info(self.future, sys.exc_info())
803  
804 if self.future is moment:
805 self.io_loop.add_callback(self.ctx_run, self.run)
806 return False
807 elif self.future is None:
808 raise Exception("no pending future")
809 elif not self.future.done():
810  
811 def inner(f: Any) -> None:
812 # Break a reference cycle to speed GC.
813 f = None # noqa: F841
814 self.ctx_run(self.run)
815  
816 self.io_loop.add_future(self.future, inner)
817 return False
818 return True
819  
820 def handle_exception(
821 self, typ: Type[Exception], value: Exception, tb: types.TracebackType
822 ) -> bool:
823 if not self.running and not self.finished:
824 self.future = Future()
825 future_set_exc_info(self.future, (typ, value, tb))
826 self.ctx_run(self.run)
827 return True
828 else:
829 return False

841 def convert_yielded(yielded: _Yieldable) -> Future:
842 """Convert a yielded object into a `.Future`.
843  
844 The default implementation accepts lists, dictionaries, and
845 Futures. This has the side effect of starting any coroutines that
846 did not start themselves, similar to `asyncio.ensure_future`.
847  
848 If the `~functools.singledispatch` library is available, this function
849 may be extended to support additional types. For example::
850  
851 @convert_yielded.register(asyncio.Future)
852 def _(asyncio_future):
853 return tornado.platform.asyncio.to_tornado_future(asyncio_future)
854  
855 .. versionadded:: 4.1
856  
857 """
858 if yielded is None or yielded is moment:
859 return moment
860 elif yielded is _null_future:
861 return _null_future
862 elif isinstance(yielded, (list, dict)):
863 return multi(yielded) # type: ignore
864 elif is_future(yielded):
865 return typing.cast(Future, yielded)
866 elif isawaitable(yielded):
867 return _wrap_awaitable(yielded) # type: ignore
868 else:
869 raise BadYieldError("yielded unknown object %r" % (yielded,))
870  
871  
872 convert_yielded = singledispatch(convert_yielded)
