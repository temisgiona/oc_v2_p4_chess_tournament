34 def test_override_set_noconvert_columns():
35 # see gh-17351
36 #
37 # Usecols needs to be sorted in _set_noconvert_columns based
38 # on the test_usecols_with_parse_dates test from test_usecols.py
39 class MyTextFileReader(TextFileReader):
40 def __init__(self):
41 self._currow = 0
42 self.squeeze = False
43  
44 class MyCParserWrapper(CParserWrapper):
45 def _set_noconvert_columns(self):
46 if self.usecols_dtype == "integer":
47 # self.usecols is a set, which is documented as unordered
48 # but in practice, a CPython set of integers is sorted.
49 # In other implementations this assumption does not hold.
50 # The following code simulates a different order, which
51 # before GH 17351 would cause the wrong columns to be
52 # converted via the parse_dates parameter
53 self.usecols = list(self.usecols)
54 self.usecols.reverse()
55 return CParserWrapper._set_noconvert_columns(self)
56

58 0,1,20140101,0900,4
59 0,1,20140102,1000,4"""

63 "a": [0, 0],
  • E501 Line too long (84 > 79 characters)

65 }
66 expected = DataFrame(cols, columns=["c_d", "a"])
