109 def flag_calls(func):
110 """Wrap a function to detect and flag when it gets called.
111  
112 This is a decorator which takes a function and wraps it in a function with
113 a 'called' attribute. wrapper.called is initialized to False.
114  
115 The wrapper.called attribute is set to False right before each call to the
116 wrapped function, so if the call fails it remains False. After the call
117 completes, wrapper.called is set to True and the output is returned.
118  
119 Testing for truth in wrapper.called allows you to determine if a call to
120 func() was attempted and succeeded."""
121  
122 # don't wrap twice
123 if hasattr(func, 'called'):
124 return func
125  
126 def wrapper(*args, **kw):
127 wrapper.called = False
128 out = func(*args, **kw)
129 wrapper.called = True
130 return out
131  
132 wrapper.called = False
133 wrapper.__doc__ = func.__doc__
134 return wrapper
