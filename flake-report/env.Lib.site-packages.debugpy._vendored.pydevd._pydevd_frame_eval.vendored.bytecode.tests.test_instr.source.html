290 jump = instr.stack_effect(jump=True)

293 self.assertEqual(instr.stack_effect(), max_effect)
294 self.assertEqual(max_effect, max(jump, no_jump))

299 for name, op in opcode.opmap.items():
300 with self.subTest(name):
301 # Use ConcreteInstr instead of Instr because it doesn't care
302 # what kind of argument it is constructed with.
303 if op < opcode.HAVE_ARGUMENT:
304 check(ConcreteInstr(name))
305 else:
306 for arg in range(256):
307 check(ConcreteInstr(name, arg))
308  
309 # LOAD_CONST uses a concrete python object as its oparg, however, in
310 # dis.stack_effect(opcode.opmap['LOAD_CONST'], oparg),
311 # oparg should be the index of that python object in the constants.
312 #
313 # Fortunately, for an instruction whose oparg isn't equivalent to its
314 # form in binary files(pyc format), the stack effect is a
315 # constant which does not depend on its oparg.
316 #
317 # The second argument of dis.stack_effect cannot be
318 # more than 2**31 - 1. If stack effect of an instruction is
319 # independent of its oparg, we pass 0 as the second argument
320 # of dis.stack_effect.
321 # (As a result we can calculate stack_effect for
322 # any LOAD_CONST instructions, even for large integers)
323  
324 for arg in 2 ** 31, 2 ** 32, 2 ** 63, 2 ** 64, -1:
325 self.assertEqual(Instr("LOAD_CONST", arg).stack_effect(), 1)
