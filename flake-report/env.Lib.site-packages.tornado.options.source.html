16 """A command line parsing module that lets modules define their own options.
17  
18 This module is inspired by Google's `gflags
19 <https://github.com/google/python-gflags>`_. The primary difference
20 with libraries such as `argparse` is that a global registry is used so
21 that options may be defined in any module (it also enables
22 `tornado.log` by default). The rest of Tornado does not depend on this
23 module, so feel free to use `argparse` or other configuration
24 libraries if you prefer them.
25  
26 Options must be defined with `tornado.options.define` before use,
27 generally at the top level of a module. The options are then
28 accessible as attributes of `tornado.options.options`::
29  
30 # myapp/db.py
31 from tornado.options import define, options
32  
33 define("mysql_host", default="127.0.0.1:3306", help="Main user DB")
34 define("memcache_hosts", default="127.0.0.1:11011", multiple=True,
35 help="Main user memcache servers")
36  
37 def connect():
38 db = database.Connection(options.mysql_host)
39 ...
40  
41 # myapp/server.py
42 from tornado.options import define, options
43  
44 define("port", default=8080, help="port to listen on")
45  
46 def start_server():
47 app = make_app()
48 app.listen(options.port)
49  
50 The ``main()`` method of your application does not need to be aware of all of
51 the options used throughout your program; they are all automatically loaded
52 when the modules are loaded. However, all modules that define options
53 must have been imported before the command line is parsed.
54  
55 Your ``main()`` method can parse the command line or parse a config file with
56 either `parse_command_line` or `parse_config_file`::
57  
58 import myapp.db, myapp.server
59 import tornado.options
60  
61 if __name__ == '__main__':
62 tornado.options.parse_command_line()
63 # or
64 tornado.options.parse_config_file("/etc/server.conf")
65  
66 .. note::
67  
68 When using multiple ``parse_*`` functions, pass ``final=False`` to all
69 but the last one, or side effects may occur twice (in particular,
70 this can result in log messages being doubled).
71  
72 `tornado.options.options` is a singleton instance of `OptionParser`, and
73 the top-level functions in this module (`define`, `parse_command_line`, etc)
74 simply call methods on it. You may create additional `OptionParser`
75 instances to define isolated sets of options, such as for subcommands.
76  
77 .. note::
78  
79 By default, several options are defined that will configure the
80 standard `logging` module when `parse_command_line` or `parse_config_file`
81 are called. If you want Tornado to leave the logging configuration
82 alone so you can manage it yourself, either pass ``--logging=none``
83 on the command line or do the following to disable it in code::
84  
85 from tornado.options import options, parse_command_line
86 options.logging = None
87 parse_command_line()
88  
89 .. versionchanged:: 4.3
90 Dashes and underscores are fully interchangeable in option names;
91 options can be defined, set, and read with any mix of the two.
92 Dashes are typical for command-line usage while config files require
93 underscores.
94 """

304 def parse_command_line(
305 self, args: Optional[List[str]] = None, final: bool = True
306 ) -> List[str]:
307 """Parses all options given on the command line (defaults to
308 `sys.argv`).
309  
310 Options look like ``--option=value`` and are parsed according
311 to their ``type``. For boolean options, ``--option`` is
312 equivalent to ``--option=true``
313  
314 If the option has ``multiple=True``, comma-separated values
315 are accepted. For multi-value integer options, the syntax
316 ``x:y`` is also accepted and equivalent to ``range(x, y)``.
317  
318 Note that ``args[0]`` is ignored since it is the program name
319 in `sys.argv`.
320  
321 We return a list of all arguments that are not parsed as options.
322  
323 If ``final`` is ``False``, parse callbacks will not be run.
324 This is useful for applications that wish to combine configurations
325 from multiple sources.
326  
327 """
328 if args is None:
329 args = sys.argv
330 remaining = [] # type: List[str]
331 for i in range(1, len(args)):
332 # All things after the last option are command line arguments
333 if not args[i].startswith("-"):
334 remaining = args[i:]
335 break
336 if args[i] == "--":
  • E203 Whitespace before ':'
337 remaining = args[i + 1 :]
338 break
339 arg = args[i].lstrip("-")
340 name, equals, value = arg.partition("=")
341 name = self._normalize_name(name)
342 if name not in self._options:
343 self.print_help()
344 raise Error("Unrecognized command line option: %r" % name)
345 option = self._options[name]
346 if not equals:
347 if option.type == bool:
348 value = "true"
349 else:
350 raise Error("Option %r requires a value" % name)
351 option.parse(value)
352  
353 if final:
354 self.run_parse_callbacks()
355  
356 return remaining

481

483 assert options.name == value

488 class _Mockable(object):
489 """`mock.patch` compatible wrapper for `OptionParser`.
490  
491 As of ``mock`` version 1.0.1, when an object uses ``__getattr__``
492 hooks instead of ``__dict__``, ``patch.__exit__`` tries to delete
493 the attribute it set instead of setting a new one (assuming that
494 the object does not capture ``__setattr__``, so the patch
495 created a new attribute in ``__dict__``).
496  
497 _Mockable's getattr and setattr pass through to the underlying
498 OptionParser, and delattr undoes the effect of a previous setattr.
499 """
500  
501 def __init__(self, options: OptionParser) -> None:
502 # Modify __dict__ directly to bypass __setattr__
503 self.__dict__["_options"] = options
504 self.__dict__["_originals"] = {}
505  
506 def __getattr__(self, name: str) -> Any:
507 return getattr(self._options, name)
508  
509 def __setattr__(self, name: str, value: Any) -> None:
510 assert name not in self._originals, "don't reuse mockable objects"
511 self._originals[name] = getattr(self._options, name)
512 setattr(self._options, name, value)
513  
514 def __delattr__(self, name: str) -> None:
515 setattr(self._options, name, self._originals.pop(name))

644 try:
