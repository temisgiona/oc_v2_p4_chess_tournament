55 def get_reply(kc, msg_id, timeout=TIMEOUT, channel='shell'):
56 t0 = time()
57 while True:
58 get_msg = getattr(kc, f'get_{channel}_msg')
59 reply = get_msg(timeout=timeout)

64 t1 = time()
65 timeout -= t1 - t0

153 def assemble_output(get_msg):
154 """assemble stdout/err from an execution"""
155 stdout = ''
156 stderr = ''
157 while True:
158 msg = get_msg(block=True, timeout=1)
159 msg_type = msg['msg_type']
160 content = msg['content']
161 if msg_type == 'status' and content['execution_state'] == 'idle':
162 # idle message signals end of output
163 break
164 elif msg['msg_type'] == 'stream':
165 if content['name'] == 'stdout':
166 stdout += content['text']
167 elif content['name'] == 'stderr':
168 stderr += content['text']
169 else:
170 raise KeyError("bad stream: %r" % content['name'])
171 else:
172 # other output, ignored
173 pass
174 return stdout, stderr
