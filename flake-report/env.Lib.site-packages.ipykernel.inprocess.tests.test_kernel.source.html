20 def _init_asyncio_patch():
21 """set default asyncio policy to be compatible with tornado
22  
23 Tornado 6 (at least) is not compatible with the default
24 asyncio implementation on Windows
25  
26 Pick the older SelectorEventLoopPolicy on Windows
27 if the known-incompatible default policy is in use.
28  
29 do this as early as possible to make it a low priority and overrideable
30  
31 ref: https://github.com/tornadoweb/tornado/issues/2608
32  
33 FIXME: if/when tornado supports the defaults in asyncio,
34 remove and bump tornado requirement for py38
35 """
36 if sys.platform.startswith("win") and sys.version_info >= (3, 8) and tornado.version_info < (6, 1):
37 import asyncio
38 try:
39 from asyncio import (
40 WindowsProactorEventLoopPolicy,
41 WindowsSelectorEventLoopPolicy,
42 )
43 except ImportError:
44 pass
45 # not affected
46 else:
47 if type(asyncio.get_event_loop_policy()) is WindowsProactorEventLoopPolicy:
48 # WindowsProactorEventLoopPolicy is not compatible with tornado 6
49 # fallback to the pre-3.8 default of Selector
50 asyncio.set_event_loop_policy(WindowsSelectorEventLoopPolicy())
