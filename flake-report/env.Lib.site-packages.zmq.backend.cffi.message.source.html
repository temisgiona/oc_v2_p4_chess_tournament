34 def _check_rc(rc):
35 err = C.zmq_errno()
36 if rc == -1:
37 if err == errno.EINTR:
38 raise zmq.error.InterrruptedSystemCall(err)
39 elif err == errno.EAGAIN:
40 raise zmq.error.Again(errno)
41 elif err == ETERM:
42 raise zmq.error.ContextTerminated(err)
43 else:
44 raise zmq.error.ZMQError(err)
