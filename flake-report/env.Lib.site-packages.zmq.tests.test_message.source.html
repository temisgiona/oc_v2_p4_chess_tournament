36 def await_gc(obj, rc):
37 """wait for refcount on an object to drop to an expected value
38  
39 Necessary because of the zero-copy gc thread,
40 which can take some time to receive its DECREF message.
41 """
42 for i in range(50):
43 # rc + 2 because of the refs in this function
44 if grc(obj) <= rc + 2:
45 return
46 time.sleep(0.05)
