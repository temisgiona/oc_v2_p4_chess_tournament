37 class PUBHandler(logging.Handler):
38 """A basic logging handler that emits log messages through a PUB socket.
39  
40 Takes a PUB socket already bound to interfaces or an interface to bind to.
41  
42 Example::
43  
44 sock = context.socket(zmq.PUB)
45 sock.bind('inproc://log')
46 handler = PUBHandler(sock)
47  
48 Or::
49  
50 handler = PUBHandler('inproc://loc')
51  
52 These are equivalent.
53  
54 Log messages handled by this handler are broadcast with ZMQ topics
55 ``this.root_topic`` comes first, followed by the log level
56 (DEBUG,INFO,etc.), followed by any additional subtopics specified in the
57 message by: log.debug("subtopic.subsub::the real message")
58 """
59  
60 socket = None

62 def __init__(self, interface_or_socket, context=None, root_topic=''):
63 logging.Handler.__init__(self)
64 self._root_topic = root_topic
65 self.formatters = {
66 logging.DEBUG: logging.Formatter(
67 "%(levelname)s %(filename)s:%(lineno)d - %(message)s\n"
68 ),
69 logging.INFO: logging.Formatter("%(message)s\n"),
70 logging.WARN: logging.Formatter(
71 "%(levelname)s %(filename)s:%(lineno)d - %(message)s\n"
72 ),
73 logging.ERROR: logging.Formatter(
74 "%(levelname)s %(filename)s:%(lineno)d - %(message)s - %(exc_info)s\n"
75 ),
76 logging.CRITICAL: logging.Formatter(
77 "%(levelname)s %(filename)s:%(lineno)d - %(message)s\n"
78 ),
79 }
80 if isinstance(interface_or_socket, zmq.Socket):
81 self.socket = interface_or_socket
82 self.ctx = self.socket.context
83 else:
84 self.ctx = context or zmq.Context()
85 self.socket = self.ctx.socket(zmq.PUB)
86 self.socket.bind(interface_or_socket)

96 def setRootTopic(self, root_topic):
97 """Set the root topic for this handler.
98  
99 This value is prepended to all messages published by this handler, and it
100 defaults to the empty string ''. When you subscribe to this socket, you must
101 set your subscription to an empty string, or to at least the first letter of
102 the binary representation of this string to ensure you receive any messages
103 from this handler.
104  
105 If you use the default empty string root topic, messages will begin with
106 the binary representation of the log level string (INFO, WARN, etc.).
107 Note that ZMQ SUB sockets can have multiple subscriptions.
108 """
109 self._root_topic = root_topic

111 def setFormatter(self, fmt, level=logging.NOTSET):
112 """Set the Formatter for this handler.
113  
114 If no level is provided, the same format is used for all levels. This
115 will overwrite all selective formatters set in the object constructor.
116 """
117 if level == logging.NOTSET:
118 for fmt_level in self.formatters.keys():
119 self.formatters[fmt_level] = fmt
120 else:
121 self.formatters[level] = fmt

123 def format(self, record):
124 """Format a record."""
125 return self.formatters[record.levelno].format(record)

127 def emit(self, record):
128 """Emit a log message on my socket."""
129 try:
130 topic, record.msg = record.msg.split(TOPIC_DELIM, 1)
131 except Exception:
132 topic = ""
133 try:
134 bmsg = cast_bytes(self.format(record))
135 except Exception:
136 self.handleError(record)
137 return
138  
139 topic_list = []
140  
141 if self.root_topic:
142 topic_list.append(self.root_topic)
143  
144 topic_list.append(record.levelname)
145  
146 if topic:
147 topic_list.append(topic)
148  
149 btopic = b'.'.join(cast_bytes(t) for t in topic_list)
150  
151 self.socket.send_multipart([btopic, bmsg])

154 class TopicLogger(logging.Logger):
155 """A simple wrapper that takes an additional argument to log methods.
156  
157 All the regular methods exist, but instead of one msg argument, two
158 arguments: topic, msg are passed.
159  
160 That is::
161  
162 logger.debug('msg')
163  
164 Would become::
165  
166 logger.debug('topic.sub', 'msg')
167 """
168  
169 def log(self, level, topic, msg, *args, **kwargs):
170 """Log 'msg % args' with level and topic.
171  
172 To pass exception information, use the keyword argument exc_info
173 with a True value::
174  
175 logger.log(level, "zmq.fun", "We have a %s",
176 "mysterious problem", exc_info=1)
177 """
178 logging.Logger.log(self, level, '%s::%s' % (topic, msg), *args, **kwargs)

181 # Generate the methods of TopicLogger, since they are just adding a
182 # topic prefix to a message.
183 for name in "debug warn warning error critical fatal".split():
184 meth = getattr(logging.Logger, name)

186 TopicLogger,
