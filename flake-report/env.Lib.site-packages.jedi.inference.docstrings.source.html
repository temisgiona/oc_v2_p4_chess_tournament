101 def _expand_typestr(type_str):
102 """
103 Attempts to interpret the possible types in `type_str`
104 """
105 # Check if alternative types are specified with 'or'
106 if re.search(r'\bor\b', type_str):
107 for t in type_str.split('or'):
108 yield t.split('of')[0].strip()
109 # Check if like "list of `type`" and set type to list
110 elif re.search(r'\bof\b', type_str):
111 yield type_str.split('of')[0]
112 # Check if type has is a set of valid literal values eg: {'C', 'F', 'A'}
113 elif type_str.startswith('{'):
114 node = parse(type_str, version='3.7').children[0]
115 if node.type == 'atom':
116 for leaf in getattr(node.children[1], "children", []):
117 if leaf.type == 'number':
118 if '.' in leaf.value:
119 yield 'float'
120 else:
121 yield 'int'
122 elif leaf.type == 'string':
123 if 'b' in leaf.string_prefix.lower():
124 yield 'bytes'
125 else:
126 yield 'str'
127 # Ignore everything else.
128  
129 # Otherwise just work with what we have.
130 else:
131 yield type_str
