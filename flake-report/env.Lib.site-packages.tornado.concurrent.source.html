75 """Decorator to run a synchronous method asynchronously on an executor.
76  
77 Returns a future.
78  
79 The executor to be used is determined by the ``executor``
80 attributes of ``self``. To use a different attribute name, pass a
81 keyword argument to the decorator::
82  
83 @run_on_executor(executor='_thread_pool')
84 def foo(self):
85 pass
86  
87 This decorator should not be confused with the similarly-named
88 `.IOLoop.run_in_executor`. In general, using ``run_in_executor``
89 when *calling* a blocking method is recommended instead of using
90 this decorator when *defining* a method. If compatibility with older
91 versions of Tornado is required, consider defining an executor
92 and using ``executor.submit()`` at the call site.
93  
94 .. versionchanged:: 4.2
95 Added keyword arguments to use alternative attributes.

144 ``b`` has already been completed or cancelled by the time ``a`` finishes.
145  
146 .. versionchanged:: 5.0
147  
148 Now accepts both Tornado/asyncio `Future` objects and
149 `concurrent.futures.Future`.

173 def future_set_result_unless_cancelled(
174 future: "Union[futures.Future[_T], Future[_T]]", value: _T
175 ) -> None:
176 """Set the given ``value`` as the `Future`'s result, if not cancelled.
177  
178 Avoids ``asyncio.InvalidStateError`` when calling ``set_result()`` on
179 a cancelled `asyncio.Future`.
180  
181 .. versionadded:: 5.0
182 """
183 if not future.cancelled():
184 future.set_result(value)

187 def future_set_exception_unless_cancelled(
188 future: "Union[futures.Future[_T], Future[_T]]", exc: BaseException
189 ) -> None:
190 """Set the given ``exc`` as the `Future`'s exception.
191  
192 If the Future is already canceled, logs the exception instead. If
193 this logging is not desired, the caller should explicitly check
194 the state of the Future and call ``Future.set_exception`` instead of
195 this wrapper.
196  
197 Avoids ``asyncio.InvalidStateError`` when calling ``set_exception()`` on
198 a cancelled `asyncio.Future`.
199  
200 .. versionadded:: 6.0
201  
202 """
203 if not future.cancelled():
204 future.set_exception(exc)
205 else:
206 app_log.error("Exception after Future was cancelled", exc_info=exc)

247 def future_add_done_callback( # noqa: F811
  • E501 Line too long (82 > 79 characters)
248 future: "Union[futures.Future[_T], Future[_T]]", callback: Callable[..., None]
249 ) -> None:
250 """Arrange to call ``callback`` when ``future`` is complete.
251  
252 ``callback`` is invoked with one argument, the ``future``.
253  
254 If ``future`` is already done, ``callback`` is invoked immediately.
255 This may differ from the behavior of ``Future.add_done_callback``,
256 which makes no such guarantee.
257  
258 .. versionadded:: 5.0
259 """
260 if future.done():
261 callback(future)
262 else:
263 future.add_done_callback(callback)
